019package org.elasticsearch.river.oai;
020
021import java.io.IOException;
022import java.net.URISyntaxException;
023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.Iterator;
026import java.util.Map;
027import org.elasticsearch.client.Client;
028import org.elasticsearch.common.xcontent.XContentBuilder;
029import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
030import org.xbib.rdf.BlankNode;
031import org.xbib.rdf.Literal;
032import org.xbib.rdf.Property;
033import org.xbib.rdf.Resource;
034import org.xbib.xml.NamespaceContext;
035import org.xbib.xml.SimpleNamespaceContext;
036
037/**
038 * Abstract class for indexing resources to Elasticsearch
039 *
040 * @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
041 */
042public abstract class AbstractWrite<S extends Resource<?, ?, ?>, P extends Property, O extends Literal<?>> {
043
044    private static final NamespaceContext context = SimpleNamespaceContext.getInstance();
045    private volatile XContentBuilder builder;
046    protected String index;
047    protected String type;
048    protected char delimiter;
049
050    public AbstractWrite(String index, String type) {
051        this(index, type, ':');
052    }
053
054    public AbstractWrite(String index, String type, char delimiter) {
055        this.delimiter = delimiter;
056        this.index = index;
057        this.type = type;
058    }
059
060    public void setIndex(String index) {
061        this.index = index;
062    }
063
064    public void setType(String type) {
065        this.type = type;
066    }
067
068    public String createId(Resource resource) {
069        if (resource.getIdentifier() == null) {
070            return null;
071        }
072        String id = resource.getIdentifier().getFragment();
073        if (id == null) {
074            id = resource.getIdentifier().toString();
075        }
076        return id;
077    }
078
079    public abstract void write(Client client, Resource<S, P, O> resource)
080            throws IOException;
081
082    public abstract void flush(Client client)
083            throws IOException;
084
085    protected void build(Resource<S, P, O> resource) throws IOException {
086        this.builder = jsonBuilder();
087        try {
088            builder.startObject();
089            build(builder, resource);
090            builder.endObject();
091        } catch (URISyntaxException e) {
092            throw new IOException(e);
093        }
094    }
095
096    protected char getDelimiter() {
097        return delimiter;
098    }
099
100    protected XContentBuilder getBuilder() {
101        return builder;
102    }
103
104    protected void build(XContentBuilder builder, Resource<S, P, O> resource)
105            throws IOException, URISyntaxException {
106        // iterate over properties
107        Iterator<P> it = resource.predicateSet(resource.getSubject()).iterator();
108        while (it.hasNext()) {
109            P predicate = it.next();
110            Collection<O> values = resource.objectSet(predicate);
111            if (values == null) {
112                throw new IllegalArgumentException(
113                        "can't build property value set for predicate URI "
114                        + predicate);
115            }
116            // drop values with size 0 silently
117            if (values.size() == 1) {
118                // single value
119                O value = values.iterator().next();
120                if (!(value instanceof BlankNode)) {
121                    builder.field(
122                            context.abbreviate(predicate.getURI()),
123                            value.toString());
124                }
125            } else if (values.size() > 1) {
126                // array of values
127                Collection<O> properties = filterBlankNodes(values, new ArrayList<O>());
128                if (!properties.isEmpty()) {
129                    builder.startArray(context.abbreviate(predicate.getURI()));
130                    for (O value : properties) {
131                        builder.value(value.toString());
132                    }
133                    builder.endArray();
134                }
135            }
136        }
137        // then, iterate over resources
138        Map<P, Collection<Resource<S, P, O>>> m = resource.resources();
139        Iterator<P> resIt = m.keySet().iterator();
140        while (resIt.hasNext()) {
141            P predicate = resIt.next();
142            Collection<Resource<S, P, O>> resources = m.get(predicate);
143            // drop resources with size 0 silently
144            if (resources.size() == 1) {
145                // single resource
146                builder.startObject(context.abbreviate(predicate.getURI()));
147                build(builder, resources.iterator().next());
148                builder.endObject();
149            } else if (resources.size() > 1) {
150                // array of resources
151                builder.startArray(context.abbreviate(predicate.getURI()));
152                for (Resource<S, P, O> child : resources) {
153                    builder.startObject();
154                    build(builder, child);
155                    builder.endObject();
156                }
157                builder.endArray();
158            }
159        }
160    }
161
162    private Collection<O> filterBlankNodes(Collection<O> objects, Collection<O> properties) {
163        for (O object : objects) {
164            if (object instanceof BlankNode) {
165                continue;
166            }
167            properties.add(object);
168        }
169        return properties;
170    }
171}

088    @Inject
089    public OAIRiver(RiverName riverName, RiverSettings settings,
090            @RiverIndexName String riverIndexName, Client client) {
091        super(riverName, settings);
092        this.riverIndexName = riverIndexName;
093        this.client = client;
094        if (settings.settings().containsKey("oai")) {
095            Map<String, Object> oaiSettings = (Map<String, Object>) settings.settings().get("oai");
096            oaiPoll = XContentMapValues.nodeTimeValue(oaiSettings.get("poll"), TimeValue.timeValueMinutes(60));
097            oaiUrl = XContentMapValues.nodeStringValue(oaiSettings.get("url"), "http://localhost");
098            oaiSet = XContentMapValues.nodeStringValue(oaiSettings.get("set"), null);
099            oaiMetadataPrefix = XContentMapValues.nodeStringValue(oaiSettings.get("metadataPrefix"), null);
100            oaiFrom = XContentMapValues.nodeStringValue(oaiSettings.get("from"), null);
101            oaiUntil = XContentMapValues.nodeStringValue(oaiSettings.get("until"), null);
102            oaiProxyHost = XContentMapValues.nodeStringValue(oaiSettings.get("proxyhost"), null);
103            oaiProxyPort = XContentMapValues.nodeIntegerValue(oaiSettings.get("proxyport"), 0);
104            oaiTimeout = XContentMapValues.nodeTimeValue(oaiSettings.get("timeout"), TimeValue.timeValueSeconds(60));
105        } else {
106            oaiPoll = TimeValue.timeValueMinutes(60);
107            oaiUrl = "http://localhost";
108            oaiSet = null;
109            oaiMetadataPrefix = null;
110            oaiFrom = null;
111            oaiUntil = null;
112            oaiProxyHost = null;
113            oaiProxyPort = 0;
114            oaiTimeout = TimeValue.timeValueSeconds(60);
115        }
116        if (settings.settings().containsKey("index")) {
117            Map<String, Object> indexSettings = (Map<String, Object>) settings.settings().get("index");
118            indexName = XContentMapValues.nodeStringValue(indexSettings.get("index"), "oai");
119            typeName = XContentMapValues.nodeStringValue(indexSettings.get("type"), "default");
120            bulkSize = XContentMapValues.nodeIntegerValue(indexSettings.get("bulk_size"), 100);
121            maxBulkRequests = XContentMapValues.nodeIntegerValue(indexSettings.get("max_bulk_requests"), 30);
122            if (indexSettings.containsKey("bulk_timeout")) {
123                bulkTimeout = TimeValue.parseTimeValue(XContentMapValues.nodeStringValue(indexSettings.get("bulk_timeout"), "60s"), TimeValue.timeValueMillis(60000));
124            } else {
125                bulkTimeout = TimeValue.timeValueMillis(60000);
126            }
127        } else {
128            indexName = "oai";
129            typeName = "default";
130            bulkSize = 100;
131            maxBulkRequests = 30;
132            bulkTimeout = TimeValue.timeValueMillis(60000);
133        }
134        operation = new BulkWrite(logger, indexName, typeName).setBulkSize(bulkSize).setMaxActiveRequests(maxBulkRequests).setMillisBeforeContinue(bulkTimeout.millis());
135    }
136
137    @Override
138    public void start() {
139        logger.info("starting OAI harvester: URL [{}], set [{}], metadataPrefix [{}], from [{}], until [{}], indexing to [{}]/[{}], poll [{}]",
140                oaiUrl, oaiSet, oaiMetadataPrefix, oaiFrom, oaiUntil, indexName, typeName, oaiPoll);
141        try {
142            client.admin().indices().prepareCreate(indexName).execute().actionGet();
143        } catch (Exception e) {
144            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
145                // that's fine
146            } else if (ExceptionsHelper.unwrapCause(e) instanceof ClusterBlockException) {
147                // ok, not recovered yet..., lets start indexing and hope we recover by the first bulk
148                // TODO: a smarter logic can be to register for cluster event listener here, and only start sampling when the block is removed...
149            } else {
150                logger.warn("failed to create index [{}], disabling river...", e, indexName);
151                return;
152            }
153        }
154        harvesterThread = EsExecutors.daemonThreadFactory(settings.globalSettings(), "oai_river_harvester").newThread(new Harvester());
155        harvesterThread.start();
156    }
157
158    @Override
159    public void close() {
160        if (closed) {
161            return;
162        }
163        logger.info("closing");
164        harvesterThread.interrupt();
165        closed = true;
166    }

231    private class Harvester implements Runnable {
232
233        private Resource resource;
234
235        public void setResource(Resource resource) {
236            this.resource = resource;
237        }
238
239        public Resource getResource() {
240            return resource;
241        }
242
243        @Override
244        public void run() {
245            final OAIClient oaiClient = OAIClientFactory.getClient(oaiUrl);
246            if (oaiProxyHost != null) {
247                oaiClient.setProxy(oaiProxyHost, oaiProxyPort);
248            }
249            if (oaiTimeout.millis() > 0) {
250                oaiClient.setTimeout(oaiTimeout.millis());
251            }
252            while (true) {
253                if (closed) {
254                    return;
255                }
256                boolean shouldHarvest = false;
257                final ListRecordsRequest request = new RiverListRecordsRequest(oaiClient.getURI());
258                request.setFrom(DateUtil.parseDateISO(oaiFrom));
259                request.setUntil(DateUtil.parseDateISO(oaiUntil));
260                try {
261                    client.admin().indices().prepareRefresh(riverIndexName).execute().actionGet();
262                    GetResponse get = client.prepareGet(riverIndexName, riverName().name(), "_last").execute().actionGet();
263                    if (!get.exists()) {
264                        shouldHarvest = true;
265                    } else {
266                        Map<String, Object> oaiState = (Map<String, Object>) get.sourceAsMap().get("oai");
267                        if (oaiState != null) {
268                            Object lastHttpStatus = oaiState.get("last_http_status");
269                            int lastHttpStatusCode = -1;
270                            if (lastHttpStatus instanceof Integer) {
271                                lastHttpStatusCode = (Integer) lastHttpStatus;
272                            }
273                            Object resumptionToken = oaiState.get("resumptionToken");
274                            if (resumptionToken != null && lastHttpStatusCode == 200) {
275                                ResumptionToken token = ResumptionToken.newToken(resumptionToken.toString());
276                                request.setResumptionToken(token);
277                                shouldHarvest = true;
278                            } else {
279                                // advance time period
280                                Object lastFrom = oaiState.get("last_from");
281                                Object lastUntil = oaiState.get("last_until");
282                                if (lastFrom != null && lastUntil != null) {
283                                    Date d1 = DateUtil.parseDateISO(lastFrom.toString());
284                                    Date d2 = DateUtil.parseDateISO(lastUntil.toString());
285                                    long delta1 = d2.getTime() - d1.getTime();
286                                    long delta2 = delta1;
287                                    // don't harvest into the future (now +/- 15 seconds)
288                                    if (d2.getTime() + delta2 < new Date().getTime() - 15000L) {
289                                        d1.setTime(d1.getTime() + delta1);
290                                        d2.setTime(d2.getTime() + delta2);
291                                        oaiFrom = DateUtil.formatDateISO(d1);
292                                        request.setFrom(d1);
293                                        oaiUntil = DateUtil.formatDateISO(d2);
294                                        request.setUntil(d2);
295                                        shouldHarvest = true;
296                                    }
297                                }
298                            }
299                        }
300                    }
301                } catch (Exception e) {
302                    logger.warn("failed to get last_OAI state, throttling", e);
303                    try {
304                        Thread.sleep(60000);
305                    } catch (InterruptedException e1) {
306                        if (closed) {
307                            return;
308                        }
309                    }
310                }
311                if (!shouldHarvest) {
312                    delay();
313                    continue;
314                }
315                logger.info("OAI harvest: URL [{}] set [{}] metadataPrefix [{}] from [{}] until [{}] resumptionToken [{}]",
316                        oaiUrl, request.getSet(), request.getMetadataPrefix(),
317                        request.getFrom(), request.getUntil(), request.getResumptionToken());
318                do {
319                    EmptyWriter w = new EmptyWriter();
320                    ListRecordsResponse response = new ListRecordsResponse(w);
321                    XmlTriplifier reader = MetadataPrefixService.getTriplifier(oaiMetadataPrefix);
322                    final StatementListener stmt = new StatementListener() {
323
324                        @Override
325                        public void newIdentifier(URI uri) {
326                            getResource().setIdentifier(uri);
327                        }
328
329                        @Override
330                        public void statement(Statement statement) {
331                            getResource().add(statement);
332                        }
333                    };
334                    reader.setListener(stmt);
335                    final XmlHandler handler = reader.getHandler();
336                    MetadataReader metadataReader = new MetadataReader() {
337
338                        @Override
339                        public void startDocument() throws SAXException {
340                            handler.startDocument();
341                            setResource(new SimpleResource());
342                        }
343
344                        @Override
345                        public void endDocument() throws SAXException {
346                            handler.endDocument();
347                            if (resource.getIdentifier() == null) {
348                                resource.setIdentifier(URI.create(getHeader().getIdentifier()));
349                            }
350                            try {
351                                operation.write(client, resource);
352                            } catch (IOException ex) {
353                                logger.error(ex.getMessage(), ex);
354                                failure = true;
355                            }
356                        }
357
358                        @Override
359                        public void startPrefixMapping(String string, String string1) throws SAXException {
360                            handler.startPrefixMapping(string, string1);
361                        }
362
363                        @Override
364                        public void endPrefixMapping(String string) throws SAXException {
365                            handler.endPrefixMapping(string);
366                        }
367
368                        @Override
369                        public void startElement(String ns, String localname, String string2, Attributes atrbts) throws SAXException {
370                            handler.startElement(ns, localname, string2, atrbts);
371                        }
372
373                        @Override
374                        public void endElement(String ns, String localname, String string2) throws SAXException {
375                            handler.endElement(ns, localname, string2);
376                        }
377
378                        @Override
379                        public void characters(char[] chars, int i, int i1) throws SAXException {
380                            handler.characters(chars, i, i1);
381                        }
382                    };
383                    oaiClient.setMetadataReader(metadataReader);
384                    int status = -1;
385                    try {
386                        StylesheetTransformer transformer = new StylesheetTransformer("xsl");
387                        oaiClient.setStylesheetTransformer(transformer);
388                        oaiClient.prepareListRecords(request, response);
389                        OAIOperation op = oaiClient.execute(oaiTimeout.getMillis(), TimeUnit.MILLISECONDS);
390                        HttpResult result = op.getResult(oaiClient.getURI());
391                        if (result == null) {
392                            logger.error("no response, failure");
393                            failure = true;
394                            closed = true;
395                        } else {
396                            status = result.getStatusCode();
397                            if (status != 200) {
398                                // transport/server errors
399                                logger.error("HTTP status = " + status);
400                                failure = true;
401                                closed = true;
402                            }
403                            if (result.getThrowable() != null) {
404                                logger.error(result.getThrowable().getMessage());
405                                failure = true;
406                                closed = true;
407                            }
408                        }
409                    } catch (IOException ex) {
410                        // conection errors
411                        logger.error(ex.getMessage(), ex);
412                        failure = true;
413                        closed = true;
414                    }
415                    // OAI errors
416                    if (response.getError() != null) {
417                        logger.error("OAI error " + response.getError());
418                        failure = true;
419                        closed = true;
420                    }
421                    // save state
422                    try {
423                        operation.flush(client);
424                        XContentBuilder builder = jsonBuilder();
425                        builder.startObject().startObject("oai");
426                        if (request.getResumptionToken() != null) {
427                            builder.field("resumptionToken", request.getResumptionToken());
428                        }
429                        if (request.getFrom() != null) {
430                            builder.field("last_from", DateUtil.formatDateISO(request.getFrom()));
431                        }
432                        if (request.getUntil() != null) {
433                            builder.field("last_until", DateUtil.formatDateISO(request.getUntil()));
434                        }
435                        builder.field("last_error", response.getError());
436                        builder.field("last_http_status", status);
437                        builder.endObject().endObject();
438                        client.prepareBulk().add(indexRequest(riverIndexName).type(riverName.name()).id("_last").source(builder)).execute().actionGet();
439                    } catch (IOException | ElasticSearchException ex) {
440                        logger.error(ex.getMessage(), ex);
441                        failure = true;
442                    }
443                } while (request.getResumptionToken() != null && !failure && !closed);
444                if (!closed) {
445                    delay();
446                }
447            }
448        }
449    }
450
451    private void delay() {
452        if (oaiPoll.millis() > 0L) {
453            logger.info("waiting {} for next run of URL [{}] set [{}] metadataPrefix [{}]",
454                    oaiPoll, oaiUrl, oaiSet, oaiMetadataPrefix);
455            try {
456                Thread.sleep(oaiPoll.millis());
457            } catch (InterruptedException e1) {
458                
459            }
460        }
461    }

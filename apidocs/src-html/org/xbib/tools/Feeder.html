001package org.xbib.tools;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
006import org.elasticsearch.common.settings.ImmutableSettings;
007import org.elasticsearch.common.unit.TimeValue;
008import org.xbib.elasticsearch.support.client.Ingest;
009import org.xbib.elasticsearch.support.client.bulk.BulkTransportClient;
010import org.xbib.elasticsearch.support.client.ingest.IngestTransportClient;
011import org.xbib.elasticsearch.support.client.ingest.MockIngestTransportClient;
012import org.xbib.pipeline.Pipeline;
013import org.xbib.pipeline.PipelineRequest;
014
015import java.io.IOException;
016import java.io.Reader;
017import java.io.Writer;
018
019public abstract class Feeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
020        extends Converter<T, R, P> {
021
022    private final static Logger logger = LogManager.getLogger(Feeder.class.getSimpleName());
023
024    protected static Ingest ingest;
025
026    @Override
027    public Feeder<T, R, P> reader(Reader reader) {
028        super.reader(reader);
029        return this;
030    }
031
032    @Override
033    public Feeder<T, R, P> writer(Writer writer) {
034        super.writer(writer);
035        return this;
036    }
037
038    protected String getIndex() {
039        return settings.get("index");
040    }
041
042    protected String getType() {
043        return settings.get("type");
044    }
045
046    protected Ingest createIngest() {
047        return settings.getAsBoolean("mock", false) ?
048                new MockIngestTransportClient() :
049                "ingest".equals(settings.get("client")) ? new IngestTransportClient() :
050                        new BulkTransportClient();
051    }
052
053    @Override
054    protected Feeder<T, R, P> prepare() throws IOException {
055        super.prepare();
056        if (ingest == null) {
057            Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
058            Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
059                    Runtime.getRuntime().availableProcessors());
060            ingest = createIngest();
061            ingest.maxActionsPerBulkRequest(maxbulkactions)
062                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests);
063        }
064        createIndex(getIndex());
065        return this;
066    }
067
068    @Override
069    protected Feeder<T, R, P> cleanup() throws IOException {
070        super.cleanup();
071        if (ingest != null) {
072            try {
073                logger.info("flush");
074                ingest.flushIngest();
075                logger.info("waiting for all responses");
076                ingest.waitForResponses(TimeValue.timeValueSeconds(120));
077            } catch (InterruptedException e) {
078                Thread.currentThread().interrupt();
079                logger.error(e.getMessage(), e);
080            }
081            logger.info("shutdown");
082            ingest.shutdown();
083        }
084        logger.info("done with run");
085        return this;
086    }
087
088    protected Feeder createIndex(String index) throws IOException {
089        ingest.newClient(ImmutableSettings.settingsBuilder()
090                .put("cluster.name", settings.get("elasticsearch.cluster"))
091                .put("host", settings.get("elasticsearch.host"))
092                .put("port", settings.getAsInt("elasticsearch.port", 9300))
093                .put("sniff", settings.getAsBoolean("elasticsearch.sniff", false))
094                .build());
095        ingest.waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
096        try {
097            beforeIndexCreation(ingest);
098            ingest.newIndex(index);
099        } catch (Exception e) {
100            if (!settings.getAsBoolean("ignoreindexcreationerror", false)) {
101                throw e;
102            } else {
103                logger.warn("index creation error, but configured to ignore");
104            }
105        }
106        return this;
107    }
108
109    protected Feeder beforeIndexCreation(Ingest output) throws IOException {
110        return this;
111    }
112}

001package org.xbib.tools;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
006import org.elasticsearch.action.admin.indices.alias.IndicesAliasesRequestBuilder;
007import org.elasticsearch.action.admin.indices.alias.get.GetAliasesResponse;
008import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequestBuilder;
009import org.elasticsearch.action.admin.indices.delete.DeleteIndexResponse;
010import org.elasticsearch.action.admin.indices.get.GetIndexResponse;
011import org.elasticsearch.common.hppc.cursors.ObjectCursor;
012import org.elasticsearch.common.joda.time.DateTime;
013import org.elasticsearch.common.joda.time.format.DateTimeFormat;
014import org.elasticsearch.common.settings.ImmutableSettings;
015import org.elasticsearch.common.unit.TimeValue;
016
017import java.io.IOException;
018import java.io.InputStream;
019import java.net.URL;
020import java.util.ArrayList;
021import java.util.List;
022import java.util.regex.Matcher;
023import java.util.regex.Pattern;
024
025public abstract class TimewindowFeeder extends Feeder {
026
027    private final static Logger logger = LogManager.getLogger(TimewindowFeeder.class.getSimpleName());
028
029    private static String index;
030
031    private static String concreteIndex;
032
033    protected void setIndex(String index) {
034        this.index = index;
035    }
036
037    protected String getIndex() {
038        return index;
039    }
040
041    protected void setConcreteIndex(String concreteIndex) {
042        this.concreteIndex = concreteIndex;
043    }
044
045    protected String getConcreteIndex() {
046        return concreteIndex;
047    }
048
049    @Override
050    protected TimewindowFeeder prepare() throws IOException {
051        if (ingest == null) {
052            Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
053            Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
054                    Runtime.getRuntime().availableProcessors());
055            ingest = createIngest();
056            ingest.maxActionsPerBulkRequest(maxbulkactions)
057                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests);
058        }
059        String timeWindow = settings.get("timewindow") != null ?
060                DateTimeFormat.forPattern(settings.get("timewindow")).print(new DateTime()) : "";
061        setConcreteIndex(resolveAlias(getIndex() + timeWindow));
062        Pattern pattern = Pattern.compile("^(.*)\\d+$");
063        Matcher m = pattern.matcher(getConcreteIndex());
064        setIndex(m.matches() ? m.group(1) : getConcreteIndex());
065        logger.info("base index name = {}, concrete index name = {}", getIndex(), getConcreteIndex());
066        super.prepare();
067        return this;
068    }
069
070    @Override
071    protected TimewindowFeeder createIndex(String index) throws IOException {
072        ingest.newClient(ImmutableSettings.settingsBuilder()
073                .put("cluster.name", settings.get("elasticsearch.cluster"))
074                .put("host", settings.get("elasticsearch.host"))
075                .put("port", settings.getAsInt("elasticsearch.port", 9300))
076                .put("sniff", settings.getAsBoolean("elasticsearch.sniff", false))
077                .build());
078        if (ingest.client() != null) {
079            ingest.waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
080            if (settings.getAsBoolean("onlyalias", false)) {
081                updateAliases();
082                return this;
083            }
084            try {
085                String indexSettings = settings.get("index-settings",
086                        "classpath:org/xbib/tools/feed/elasticsearch/settings.json");
087                InputStream indexSettingsInput = (indexSettings.startsWith("classpath:") ?
088                        new URL(null, indexSettings, new ClasspathURLStreamHandler()) :
089                        new URL(indexSettings)).openStream();
090                String indexMappings = settings.get("index-mapping",
091                        "classpath:org/xbib/tools/feed/elasticsearch/mapping.json");
092                InputStream indexMappingsInput = (indexMappings.startsWith("classpath:") ?
093                        new URL(null, indexMappings, new ClasspathURLStreamHandler()) :
094                        new URL(indexMappings)).openStream();
095                ingest.newIndex(getConcreteIndex(), getType(),
096                        indexSettingsInput, indexMappingsInput);
097                indexSettingsInput.close();
098                indexMappingsInput.close();
099                ingest.startBulk(getConcreteIndex());
100            } catch (Exception e) {
101                if (!settings.getAsBoolean("ignoreindexcreationerror", false)) {
102                    throw e;
103                } else {
104                    logger.warn("index creation error, but configured to ignore", e);
105                }
106            }
107        }
108        return this;
109    }
110
111    protected String resolveAlias(String alias) {
112        if (ingest.client() == null) {
113            return alias;
114        }
115        GetAliasesResponse getAliasesResponse = ingest.client().admin().indices().prepareGetAliases(alias).execute().actionGet();
116        if (!getAliasesResponse.getAliases().isEmpty()) {
117            return getAliasesResponse.getAliases().keys().iterator().next().value;
118        }
119        return alias;
120    }
121
122    protected void updateAliases() {
123        if (ingest.client() == null) {
124            return;
125        }
126        String index = getIndex();
127        String concreteIndex = getConcreteIndex();
128        if (!index.equals(concreteIndex)) {
129            IndicesAliasesRequestBuilder requestBuilder = ingest.client().admin().indices().prepareAliases();
130            GetAliasesResponse getAliasesResponse = ingest.client().admin().indices().prepareGetAliases(index).execute().actionGet();
131            if (getAliasesResponse.getAliases().isEmpty()) {
132                logger.info("adding alias {} to index {}", index, concreteIndex);
133                requestBuilder.addAlias(concreteIndex, index);
134                // identifier is alias
135                if (settings.get("identifier") != null) {
136                    requestBuilder.addAlias(concreteIndex, settings.get("identifier"));
137                }
138            } else {
139                for (ObjectCursor<String> indexName : getAliasesResponse.getAliases().keys()) {
140                    if (indexName.value.startsWith(index)) {
141                        logger.info("switching alias {} from index {} to index {}", index, indexName.value, concreteIndex);
142                        requestBuilder.removeAlias(indexName.value, index)
143                                .addAlias(concreteIndex, index);
144                        if (settings.get("identifier") != null) {
145                            requestBuilder.removeAlias(indexName.value, settings.get("identifier"))
146                                    .addAlias(concreteIndex, settings.get("identifier"));
147                        }
148                    }
149                }
150            }
151            requestBuilder.execute().actionGet();
152            if (settings.getAsBoolean("retention.enabled", false)) {
153                performRetentionPolicy(
154                        getIndex(),
155                        getConcreteIndex(),
156                        settings.getAsInt("retention.diff", 48),
157                        settings.getAsInt("retention.mintokeep", 2));
158            }
159        }
160    }
161
162    public void performRetentionPolicy(String index, String concreteIndex, int timestampdiff, int mintokeep) {
163        if (ingest.client() == null) {
164            return;
165        }
166        if (index.equals(concreteIndex)) {
167            return;
168        }
169        GetIndexResponse getIndexResponse = ingest.client().admin().indices()
170                .prepareGetIndex()
171                .execute().actionGet();
172        Pattern pattern = Pattern.compile("^(.*?)(\\d+)$");
173        List<String> indices = new ArrayList<>();
174        logger.info("{} indices", getIndexResponse.getIndices().length);
175        for (String s : getIndexResponse.getIndices()) {
176            Matcher m = pattern.matcher(s);
177            if (m.matches()) {
178                if (index.equals(m.group(1)) && !s.equals(concreteIndex)) {
179                    indices.add(s);
180                }
181            }
182        }
183        if (indices.isEmpty()) {
184            logger.info("no indices found, retention policy skipped");
185            return;
186        }
187        if (mintokeep > 0 && indices.size() < mintokeep) {
188            logger.info("{} indices found, not enough for retention policy ({}),  skipped",
189                    indices.size(), mintokeep);
190            return;
191        } else {
192            logger.info("candidates for deletion = {}", indices);
193        }
194        List<String> indicesToDelete = new ArrayList<String>();
195        // our index
196        Matcher m1 = pattern.matcher(concreteIndex);
197        if (m1.matches()) {
198            Integer i1 = Integer.parseInt(m1.group(2));
199            for (String s : indices) {
200                Matcher m2 = pattern.matcher(s);
201                if (m2.matches()) {
202                    Integer i2 = Integer.parseInt(m2.group(2));
203                    int kept = 1 + indices.size() - indicesToDelete.size();
204                    if (timestampdiff > 0 && i1 - i2 > timestampdiff && mintokeep <= kept) {
205                        indicesToDelete.add(s);
206                    }
207                }
208            }
209        }
210        logger.info("indices to delete = {}", indicesToDelete);
211        if (indicesToDelete.isEmpty()) {
212            logger.info("not enough indices found to delete, retention policy complete");
213            return;
214        }
215        String[] s = indicesToDelete.toArray(new String[indicesToDelete.size()]);
216        DeleteIndexRequestBuilder requestBuilder = ingest.client().admin().indices().prepareDelete(s);
217        DeleteIndexResponse response = requestBuilder.execute().actionGet();
218        if (!response.isAcknowledged()) {
219            logger.warn("retention delete index operation was not acknowledged");
220        }
221    }

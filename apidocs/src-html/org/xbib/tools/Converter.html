001package org.xbib.tools;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.xbib.common.settings.Settings;
006import org.xbib.io.archive.file.Finder;
007import org.xbib.metrics.MeterMetric;
008import org.xbib.pipeline.AbstractPipeline;
009import org.xbib.pipeline.Pipeline;
010import org.xbib.pipeline.PipelineException;
011import org.xbib.pipeline.PipelineProvider;
012import org.xbib.pipeline.PipelineRequest;
013import org.xbib.pipeline.element.URIPipelineElement;
014import org.xbib.pipeline.simple.MetricSimplePipelineExecutor;
015
016import java.io.IOException;
017import java.io.Reader;
018import java.io.Writer;
019import java.net.URI;
020import java.util.Queue;
021import java.util.concurrent.ConcurrentLinkedQueue;
022
023import static org.xbib.common.settings.ImmutableSettings.settingsBuilder;
024
025public abstract class Converter<T, R extends PipelineRequest, P extends Pipeline<T, R>>
026        extends AbstractPipeline<URIPipelineElement, PipelineException> implements CommandLineInterpreter {
027
028    private final static Logger logger = LogManager.getLogger(Converter.class.getSimpleName());
029
030    private final static URIPipelineElement uriPipelineElement = new URIPipelineElement();
031
032    protected Reader reader;
033
034    protected Writer writer;
035
036    protected static Settings settings;
037
038    protected static Queue<URI> input;
039
040    protected MetricSimplePipelineExecutor<T, R, P> executor;
041
042    private boolean done = false;
043
044    public Converter<T, R, P> reader(Reader reader) {
045        this.reader = reader;
046        settings = settingsBuilder().loadFromReader(reader).build();
047        return this;
048    }
049
050    public Converter<T, R, P> writer(Writer writer) {
051        this.writer = writer;
052        return this;
053    }
054
055    protected Converter<T, R, P> prepare() throws IOException {
056        if (settings.get("uri") != null) {
057            input = new ConcurrentLinkedQueue<URI>();
058            input.add(URI.create(settings.get("uri")));
059            // parallel URI connection possible?
060            if (settings.getAsBoolean("parallel", false)) {
061                for (int i = 1; i < settings.getAsInt("concurrency", 1); i++) {
062                    input.add(URI.create(settings.get("uri")));
063                }
064            }
065        } else {
066            input = new Finder(settings.get("pattern"))
067                    .find(settings.get("path"))
068                    .pathSorted(settings.getAsBoolean("isPathSorted", false))
069                    .chronologicallySorted(settings.getAsBoolean("isChronologicallySorted", false))
070                    .getURIs();
071        }
072        logger.info("input = {}", input);
073        return this;
074    }
075
076    public void run() throws Exception {
077        try {
078            logger.info("preparing with settings {}", settings.getAsMap());
079            prepare();
080            logger.info("executing");
081            //metric pipeline setExecutor only uses concurrency over different URIs
082            // in the input queue, not with a single URI input
083            executor = new MetricSimplePipelineExecutor<T, R, P>()
084                    .setConcurrency(settings.getAsInt("concurrency", 1))
085                    .setPipelineProvider(pipelineProvider())
086                    .prepare()
087                    .execute()
088                    .waitFor();
089            logger.info("execution completed");
090        } finally {
091            cleanup();
092            if (executor != null) {
093                executor.shutdown();
094            }
095        }
096    }
097
098    public Converter<T, R, P> run(Settings newSettings, Queue<URI> newInput) throws Exception {
099        try {
100            settings = newSettings;
101            input = newInput;
102            logger.info("executing with settings {} and input {}", settings, input);
103            executor = new MetricSimplePipelineExecutor<T, R, P>()
104                    .setConcurrency(settings.getAsInt("concurrency", 1))
105                    .setPipelineProvider(pipelineProvider())
106                    .prepare();
107            executor.execute()
108                    .waitFor();
109            logger.info("execution completed");
110        } finally {
111            cleanup();
112            if (executor != null) {
113                executor.shutdown();
114            }
115        }
116        return this;
117    }
118
119    protected Converter<T, R, P> cleanup() throws IOException {
120        return this;
121    }
122
123    @Override
124    public void close() throws IOException {
125        logger.info("pipeline close (no op)");
126    }
127
128    @Override
129    public boolean hasNext() {
130        return !done && !input.isEmpty();
131    }
132
133    @Override
134    public URIPipelineElement next() {
135        URI uri = input.poll();
136        done = uri == null;
137        uriPipelineElement.set(uri);
138        if (done) {
139            logger.info("done is true");
140            return uriPipelineElement;
141        }
142        return uriPipelineElement;
143    }
144
145    @Override
146    public void newRequest(Pipeline<MeterMetric, URIPipelineElement> pipeline, URIPipelineElement request) {
147        try {
148            process(request.get());
149        } catch (Exception ex) {
150            logger.error("error while getting next input: " + ex.getMessage(), ex);
151        }
152    }
153
154    @Override
155    public void error(Pipeline<MeterMetric, URIPipelineElement> pipeline, URIPipelineElement request, PipelineException error) {
156        logger.error(error.getMessage(), error);
157    }
158
159    protected abstract PipelineProvider<P> pipelineProvider();
160
161    protected abstract void process(URI uri) throws Exception;
162
163}

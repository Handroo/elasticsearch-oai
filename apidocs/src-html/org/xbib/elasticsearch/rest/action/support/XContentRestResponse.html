002package org.xbib.elasticsearch.rest.action.support;
003
004import org.apache.lucene.util.BytesRef;
005import org.apache.lucene.util.UnicodeUtil;
006
007import org.elasticsearch.ElasticsearchIllegalArgumentException;
008import org.elasticsearch.rest.AbstractRestResponse;
009import org.elasticsearch.rest.RestRequest;
010import org.elasticsearch.rest.RestStatus;
011
012import org.xbib.common.xcontent.XContentBuilder;
013
014import java.io.IOException;
015
016public class XContentRestResponse extends AbstractRestResponse {
017
018    private static final byte[] END_JSONP;
019
020    static {
021        BytesRef U_END_JSONP = new BytesRef();
022        UnicodeUtil.UTF16toUTF8(");", 0, ");".length(), U_END_JSONP);
023        END_JSONP = new byte[U_END_JSONP.length];
024        System.arraycopy(U_END_JSONP.bytes, U_END_JSONP.offset, END_JSONP, 0, U_END_JSONP.length);
025    }
026
027    private final BytesRef prefixUtf8Result;
028
029    private final RestStatus status;
030
031    private final XContentBuilder builder;
032
033    public XContentRestResponse(RestRequest request, RestStatus status, XContentBuilder builder) throws IOException {
034        if (request == null) {
035            throw new ElasticsearchIllegalArgumentException("request must be set");
036        }
037        this.builder = builder;
038        this.status = status;
039        this.prefixUtf8Result = startJsonp(request);
040    }
041
042    public XContentBuilder builder() {
043        return this.builder;
044    }
045
046    @Override
047    public String contentType() {
048        return builder.contentType().restContentType();
049    }
050
051    @Override
052    public boolean contentThreadSafe() {
053        return true;
054    }
055
056    @Override
057    public byte[] content() throws IOException {
058        return builder.bytes().array();
059    }
060
061    @Override
062    public int contentLength() throws IOException {
063        return builder.bytes().length();
064    }
065
066    @Override
067    public int contentOffset() throws IOException {
068        return 0;
069    }
070
071    @Override
072    public RestStatus status() {
073        return this.status;
074    }
075
076    @Override
077    public byte[] prefixContent() {
078        if (prefixUtf8Result != null) {
079            return prefixUtf8Result.bytes;
080        }
081        return null;
082    }
083
084    @Override
085    public int prefixContentLength() {
086        if (prefixUtf8Result != null) {
087            return prefixUtf8Result.length;
088        }
089        return 0;
090    }
091
092    @Override
093    public int prefixContentOffset() {
094        if (prefixUtf8Result != null) {
095            return prefixUtf8Result.offset;
096        }
097        return 0;
098    }
099
100    @Override
101    public byte[] suffixContent() {
102        if (prefixUtf8Result != null) {
103            return END_JSONP;
104        }
105        return null;
106    }
107
108    @Override
109    public int suffixContentLength() {
110        if (prefixUtf8Result != null) {
111            return END_JSONP.length;
112        }
113        return 0;
114    }
115
116    @Override
117    public int suffixContentOffset() {
118        return 0;
119    }
120
121    private static BytesRef startJsonp(RestRequest request) {
122        String callback = request.param("callback");
123        if (callback == null) {
124            return null;
125        }
126        final BytesRef result = new BytesRef();
127        UnicodeUtil.UTF16toUTF8(callback, 0, callback.length(), result);
128        result.bytes[result.length] = '(';
129        result.length++;
130        return result;
131    }
132}

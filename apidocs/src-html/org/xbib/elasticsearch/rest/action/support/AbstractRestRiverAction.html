002package org.xbib.elasticsearch.rest.action.support;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.collect.ImmutableMap;
006import org.elasticsearch.common.inject.Injector;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.BaseRestHandler;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestRequest;
011import org.elasticsearch.rest.RestStatus;
012import org.elasticsearch.river.River;
013import org.elasticsearch.river.RiverName;
014import org.elasticsearch.river.RiversService;
015
016import org.xbib.common.xcontent.XContentBuilder;
017
018import java.io.IOException;
019import java.lang.reflect.Field;
020import java.util.Collections;
021
022public abstract class AbstractRestRiverAction extends BaseRestHandler {
023
024    protected Injector injector;
025
026    public AbstractRestRiverAction(Settings settings, Client client, Injector injector) {
027        super(settings, client);
028        this.injector = injector;
029    }
030
031    protected void respond(boolean success, RestRequest request, RestChannel channel, String error, RestStatus status) {
032        try {
033            XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
034            if (request.paramAsBoolean("pretty", false)) {
035                builder.prettyPrint();
036            }
037            builder.startObject()
038                 .field("success", success);
039            if (error != null) {
040                builder.field("error", error);
041            }
042            builder.endObject();
043            channel.sendResponse(new XContentRestResponse(request, status, builder));
044        } catch (IOException e) {
045            errorResponse(request, channel, e);
046        }
047    }
048
049    protected void errorResponse(RestRequest request, RestChannel channel, Throwable e) {
050        try {
051            channel.sendResponse(new XContentThrowableRestResponse(request, e));
052        } catch (IOException e1) {
053            logger.error("Failed to send failure response", e1);
054        }
055    }
056
057    /**
058     * Retrieve the registered rivers using reflection (UGLY HACK!!)
059     * @param injector injector
060     * @return map of rivers or null if not possible
061     */
062    protected ImmutableMap<RiverName, River> rivers(Injector injector) {
063        RiversService riversService = injector.getInstance(RiversService.class);
064        try {
065            Field field = RiversService.class.getDeclaredField("rivers");
066            if (field != null) {
067                field.setAccessible(true);
068                return (ImmutableMap<RiverName, River>) field.get(riversService);
069            }
070        } catch (Throwable e) {
071            //
072        }
073        return ImmutableMap.copyOf(Collections.emptyMap());
074    }
075}

002package org.xbib.elasticsearch.rest.action.river;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.inject.Injector;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.RestChannel;
009import org.elasticsearch.rest.RestController;
010import org.elasticsearch.rest.RestRequest;
011import org.elasticsearch.rest.RestStatus;
012
013import org.xbib.common.xcontent.ToXContent;
014import org.xbib.common.xcontent.XContentBuilder;
015import org.xbib.elasticsearch.plugin.river.StatefulRiver;
016import org.xbib.elasticsearch.rest.action.support.XContentRestResponse;
017import org.xbib.elasticsearch.rest.action.support.AbstractRestRiverAction;
018import org.xbib.elasticsearch.rest.action.support.XContentThrowableRestResponse;
019
020import java.io.IOException;
021
022import static org.xbib.common.xcontent.XContentFactory.jsonBuilder;
023
024public class RestRiverStateAction extends AbstractRestRiverAction {
025
026    @Inject
027    public RestRiverStateAction(Settings settings, Client client,
028                                RestController controller,
029                                Injector injector) {
030        super(settings, client, injector);
031        controller.registerHandler(RestRequest.Method.GET, "/_river/_state", this);
032        controller.registerHandler(RestRequest.Method.GET, "/_river/{river}/_state", this);
033    }
034
035    @Override
036    public void handleRequest(RestRequest request, RestChannel channel) {
037        try {
038            String riverName = request.param("river");
039            XContentBuilder builder = jsonBuilder();
040            if (request.paramAsBoolean("pretty", false)) {
041                builder.prettyPrint().lfAtEnd();
042            }
043            builder.startObject().startArray("result");
044            rivers(injector).entrySet().stream()
045                    .filter(entry -> (riverName == null || entry.getKey().getName().equals(riverName)) && (entry.getValue() instanceof StatefulRiver))
046                    .forEach(entry -> {
047                StatefulRiver river = (StatefulRiver) entry.getValue();
048                try {
049                    builder.startObject().field("state");
050                    river.getRiverState().toXContent(builder, ToXContent.EMPTY_PARAMS);
051                    builder.endObject();
052                } catch (IOException e) {
053                    // ignore
054                }
055            });
056            builder.endArray().endObject();
057            channel.sendResponse(new XContentRestResponse(request, RestStatus.OK, builder));
058        } catch (IOException ioe) {
059            try {
060                channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
061            } catch (IOException e) {
062                logger.error("unable to send response to client");
063            }
064        }
065    }
066
067}

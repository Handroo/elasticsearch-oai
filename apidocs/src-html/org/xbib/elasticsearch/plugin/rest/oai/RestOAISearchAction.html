001package org.xbib.elasticsearch.plugin.rest.oai;
002
003import org.elasticsearch.ElasticsearchIllegalStateException;
004import org.elasticsearch.action.search.SearchRequestBuilder;
005import org.elasticsearch.client.Client;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.BaseRestHandler;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestController;
011import org.elasticsearch.rest.RestRequest;
012import org.xbib.elasticsearch.oai.OAIServer;
013import org.xbib.oai.OAIDateResolution;
014import org.xbib.oai.exceptions.OAIException;
015import org.xbib.oai.server.listrecords.ListRecordsServerRequest;
016import org.xbib.oai.util.ResumptionToken;
017import org.xbib.util.DateUtil;
018
019import java.util.Date;
020
021import static org.elasticsearch.rest.RestRequest.Method.GET;
022import static org.elasticsearch.rest.RestRequest.Method.POST;
023
024public class RestOAISearchAction extends BaseRestHandler {
025
026    private final OAIServer oaiServer;
027
028    @Inject
029    public RestOAISearchAction(Settings settings, Client client, RestController controller,
030                               OAIServer oaiServer) {
031        super(settings, controller, client);
032        this.oaiServer = oaiServer;
033        controller.registerHandler(GET, "/_oai", this);
034        controller.registerHandler(POST, "/_oai", this);
035        controller.registerHandler(GET, "/{index}/_oai", this);
036        controller.registerHandler(POST, "/{index}/_oai", this);
037        controller.registerHandler(GET, "/{index}/{type}/_oai", this);
038        controller.registerHandler(POST, "/{index}/{type}/_oai", this);
039        controller.registerHandler(GET, "/_oai/template", this);
040        controller.registerHandler(POST, "/_search_arrayformat/template", this);
041        controller.registerHandler(GET, "/{index}/_search_arrayformat/template", this);
042        controller.registerHandler(POST, "/{index}/_search_arrayformat/template", this);
043        controller.registerHandler(GET, "/{index}/{type}/_search_arrayformat/template", this);
044        controller.registerHandler(POST, "/{index}/{type}/_search_arrayformat/template", this);
045    }
046
047    @Override
048    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
049        String verb = oaiServer.verbOf(request);
050        OAIResponseBuilder builder = new OAIResponseBuilder(settings, channel);
051        try {
052            switch (verb) {
053                case "ListRecords": {
054                    SearchRequestBuilder searchRequestBuilder = new SearchRequestBuilder(client);
055                    oaiServer.listRecords(makeListRecordsRequest(request), searchRequestBuilder);
056                    client.search(searchRequestBuilder.request(), builder);
057                    break;
058                }
059                default: {
060                    builder.onFailure(new ElasticsearchIllegalStateException("unknown verb: " + verb));
061                    break;
062                }
063            }
064        } catch (OAIException e) {
065            builder.onFailure(e);
066        }
067    }
068
069    private ListRecordsServerRequest makeListRecordsRequest(RestRequest restRequest) {
070        Date from = DateUtil.parseDateISO(restRequest.param("from"));
071        Date until = DateUtil.parseDateISO(restRequest.param("until"));
072        String set = restRequest.param("set");
073        String metadataPrefix = restRequest.param("metadataPrefix");
074        ResumptionToken token = ResumptionToken.newToken(restRequest.param("resumptionToken"));
075        return new ListRecordsServerRequest()
076                .setResumptionToken(token)
077                .setFrom(from, OAIDateResolution.SECOND)
078                .setUntil(until, OAIDateResolution.SECOND)
079                .setSet(set)
080                .setMetadataPrefix(metadataPrefix);
081    }
082
083}

001package org.xbib.elasticsearch.plugin.rest.oai;
002
003import org.elasticsearch.action.search.SearchResponse;
004import org.elasticsearch.common.io.stream.BytesStreamOutput;
005import org.elasticsearch.common.joda.time.format.DateTimeFormatter;
006import org.elasticsearch.common.joda.time.format.ISODateTimeFormat;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.BytesRestResponse;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestResponse;
011import org.elasticsearch.rest.RestStatus;
012import org.elasticsearch.rest.action.support.RestResponseListener;
013import org.elasticsearch.search.SearchHit;
014import org.xbib.oai.OAIConstants;
015import org.xbib.oai.server.ServerOAIRequest;
016import org.xbib.xml.XSI;
017
018import javax.xml.namespace.QName;
019import javax.xml.stream.XMLEventFactory;
020import javax.xml.stream.XMLOutputFactory;
021import javax.xml.stream.XMLStreamException;
022import javax.xml.stream.util.XMLEventConsumer;
023import java.io.IOException;
024import java.io.OutputStream;
025import java.util.Arrays;
026import java.util.Date;
027import java.util.List;
028import java.util.Map;
029
030public class OAIResponseBuilder extends RestResponseListener<SearchResponse> implements OAIConstants {
031
032    private final static XMLEventFactory eventFactory = XMLEventFactory.newInstance();
033
034    private final static XMLOutputFactory outputfactory  = XMLOutputFactory.newInstance();
035
036    private final static DateTimeFormatter formatter = ISODateTimeFormat.dateTimeNoMillis();
037
038    private final Settings settings;
039
040    private XMLEventConsumer consumer;
041
042    public OAIResponseBuilder(Settings settings, RestChannel channel) {
043        super(channel);
044        this.settings = settings;
045    }
046
047    protected Settings settings() {
048        return settings;
049    }
050
051    @Override
052    public RestResponse buildResponse(SearchResponse response) throws Exception {
053        return buildResponse(response, channel.bytesOutput());
054    }
055
056    protected RestResponse buildResponse(SearchResponse response, BytesStreamOutput out) throws Exception {
057        if (response.getHits().getHits().length > 0) {
058            for (SearchHit hits : response.getHits()) {
059                // TODO
060            }
061        }
062        return new BytesRestResponse(RestStatus.OK, "text/xml; charset=UTF-8", out.bytes(), true);
063    }
064
065    protected OAIResponseBuilder setOutputStream(OutputStream out) throws XMLStreamException {
066        this.consumer = outputfactory.createXMLEventWriter(out);
067        return this;
068    }
069
070    protected OAIResponseBuilder beginDocument() throws XMLStreamException {
071        consumer.add(eventFactory.createStartDocument());
072        return this;
073    }
074
075    protected OAIResponseBuilder endDocument() throws XMLStreamException, IOException {
076        consumer.add(eventFactory.createEndDocument());
077        return this;
078    }
079
080    protected OAIResponseBuilder beginElement(String name) throws XMLStreamException {
081        consumer.add(eventFactory.createStartElement(toQName(NS_URI, name), null, null));
082        return this;
083    }
084
085    protected OAIResponseBuilder endElement(String name) throws XMLStreamException {
086        consumer.add(eventFactory.createEndElement(toQName(NS_URI, name), null));
087        return this;
088    }
089
090    protected OAIResponseBuilder element(String name) throws XMLStreamException {
091        consumer.add(eventFactory.createStartElement(toQName(NS_URI, name), null, null));
092        consumer.add(eventFactory.createEndElement(toQName(NS_URI, name), null));
093        return this;
094    }
095
096    protected OAIResponseBuilder element(String name, String value) throws XMLStreamException {
097        consumer.add(eventFactory.createStartElement(toQName(NS_URI, name), null, null));
098        consumer.add(eventFactory.createCharacters(value));
099        consumer.add(eventFactory.createEndElement(toQName(NS_URI, name), null));
100        return this;
101    }
102
103    protected OAIResponseBuilder element(String name, Date value) throws XMLStreamException {
104        consumer.add(eventFactory.createStartElement(toQName(NS_URI, name), null, null));
105        consumer.add(eventFactory.createCharacters(formatter.print(value.getTime())));
106        consumer.add(eventFactory.createEndElement(toQName(NS_URI, name), null));
107        return this;
108    }
109
110    protected OAIResponseBuilder beginResponse(String baseURL, ServerOAIRequest serverOAIRequest) throws XMLStreamException {
111        consumer.add(eventFactory.createStartElement(toQName(NS_URI, "OAI-PMH"), null, null));
112        consumer.add(eventFactory.createNamespace(NS_URI));
113        consumer.add(eventFactory.createNamespace(XSI.NS_PREFIX, XSI.NS_URI));
114        consumer.add(eventFactory.createAttribute(XSI.NS_PREFIX, XSI.NS_URI,
115                "schemaLocation", "http://www.openarchives.org/OAI/2.0/ http://www.openarchives.org/OAI/2.0/OAI-PMH.xsd"));
116        element("responseDate", new Date());
117        request(serverOAIRequest.getParameterMap(), baseURL);
118        return this;
119    }
120
121    protected OAIResponseBuilder endResponse() throws XMLStreamException {
122        consumer.add(eventFactory.createEndElement(toQName(NS_URI, "OAI-PMH"), null));
123        return this;
124    }
125
126    protected OAIResponseBuilder request(Map<String,Object> attrs, String baseURL) throws XMLStreamException {
127        consumer.add(eventFactory.createStartElement(toQName(NS_URI, REQUEST), null, null));
128        for (Map.Entry<String,Object> me : attrs.entrySet()) {
129            Object o = me.getValue();
130            if (!(o instanceof List)) {
131                o = Arrays.asList(o);
132            }
133            for (Object value : (List)o) {
134                consumer.add(eventFactory.createAttribute(me.getKey(),value.toString()));
135            }
136        }
137        consumer.add(eventFactory.createCharacters(baseURL));
138        consumer.add(eventFactory.createEndElement(toQName(NS_URI, REQUEST), null));
139        return this;
140    }
141
142    protected QName toQName(String namespaceUri, String qname) {
143        int i = qname.indexOf(':');
144        if (i == -1) {
145            return new QName(namespaceUri, qname);
146        } else {
147            String prefix = qname.substring(0, i);
148            String localPart = qname.substring(i + 1);
149            return new QName(namespaceUri, localPart, prefix);
150        }
151    }
152}

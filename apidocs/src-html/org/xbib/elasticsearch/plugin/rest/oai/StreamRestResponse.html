002package org.xbib.elasticsearch.plugin.rest.oai;
003
004import org.elasticsearch.ElasticsearchException;
005import org.elasticsearch.common.bytes.BytesArray;
006import org.elasticsearch.common.bytes.BytesReference;
007import org.elasticsearch.common.xcontent.XContentBuilder;
008import org.elasticsearch.rest.RestChannel;
009import org.elasticsearch.rest.RestRequest;
010import org.elasticsearch.rest.RestResponse;
011import org.elasticsearch.rest.RestStatus;
012
013import java.io.IOException;
014
015import static org.elasticsearch.ExceptionsHelper.detailedMessage;
016
017public class StreamRestResponse extends RestResponse {
018
019    public static final String TEXT_CONTENT_TYPE = "text/plain; charset=UTF-8";
020
021    private final RestStatus status;
022    private final BytesReference content;
023    private final boolean contentThreadSafe;
024    private final String contentType;
025
026    public StreamRestResponse(RestStatus status) {
027        this(status, TEXT_CONTENT_TYPE, BytesArray.EMPTY, true);
028    }
029
030    /**
031     * Creates a new response based on {@link org.elasticsearch.common.xcontent.XContentBuilder}.
032     */
033    public StreamRestResponse(RestStatus status, XContentBuilder builder) {
034        this(status, builder.contentType().restContentType(), builder.bytes(), true);
035    }
036
037    /**
038     * Creates a new plain text response.
039     */
040    public StreamRestResponse(RestStatus status, String content) {
041        this(status, TEXT_CONTENT_TYPE, new BytesArray(content), true);
042    }
043
044    /**
045     * Creates a new plain text response.
046     */
047    public StreamRestResponse(RestStatus status, String contentType, String content) {
048        this(status, contentType, new BytesArray(content), true);
049    }
050
051    /**
052     * Creates a binary response.
053     */
054    public StreamRestResponse(RestStatus status, String contentType, byte[] content) {
055        this(status, contentType, new BytesArray(content), true);
056    }
057
058    /**
059     * Creates a binary response.
060     */
061    public StreamRestResponse(RestStatus status, String contentType, BytesReference content, boolean contentThreadSafe) {
062        this.status = status;
063        this.content = content;
064        this.contentThreadSafe = contentThreadSafe;
065        this.contentType = contentType;
066    }
067
068    public StreamRestResponse(RestChannel channel, Throwable t) throws IOException {
069        this(channel, ((t instanceof ElasticsearchException) ? ((ElasticsearchException) t).status() : RestStatus.INTERNAL_SERVER_ERROR), t);
070    }
071
072    public StreamRestResponse(RestChannel channel, RestStatus status, Throwable t) throws IOException {
073        this.status = status;
074        if (channel.request().method() == RestRequest.Method.HEAD) {
075            this.content = BytesArray.EMPTY;
076            this.contentType = TEXT_CONTENT_TYPE;
077        } else {
078            XContentBuilder builder = convert(channel, status, t);
079            this.content = builder.bytes();
080            this.contentType = builder.contentType().restContentType();
081        }
082        this.contentThreadSafe = true;
083    }
084
085    @Override
086    public String contentType() {
087        return this.contentType;
088    }
089
090    @Override
091    public boolean contentThreadSafe() {
092        return this.contentThreadSafe;
093    }
094
095    @Override
096    public BytesReference content() {
097        return this.content;
098    }
099
100    @Override
101    public RestStatus status() {
102        return this.status;
103    }
104
105    private static XContentBuilder convert(RestChannel channel, RestStatus status, Throwable t) throws IOException {
106        XContentBuilder builder = channel.newBuilder().startObject()
107                .field("error", detailedMessage(t))
108                .field("status", status.getStatus());
109        if (t != null && channel.request().paramAsBoolean("error_trace", false)) {
110            builder.startObject("error_trace");
111            boolean first = true;
112            int counter = 0;
113            while (t != null) {
114                // bail if there are more than 10 levels, becomes useless really...
115                if (counter++ > 10) {
116                    break;
117                }
118                if (!first) {
119                    builder.startObject("cause");
120                }
121                buildThrowable(t, builder);
122                if (!first) {
123                    builder.endObject();
124                }
125                t = t.getCause();
126                first = false;
127            }
128            builder.endObject();
129        }
130        builder.endObject();
131        return builder;
132    }
133
134    private static void buildThrowable(Throwable t, XContentBuilder builder) throws IOException {
135        builder.field("message", t.getMessage());
136        for (StackTraceElement stElement : t.getStackTrace()) {
137            builder.startObject("at")
138                    .field("class", stElement.getClassName())
139                    .field("method", stElement.getMethodName());
140            if (stElement.getFileName() != null) {
141                builder.field("file", stElement.getFileName());
142            }
143            if (stElement.getLineNumber() >= 0) {
144                builder.field("line", stElement.getLineNumber());
145            }
146            builder.endObject();
147        }
148    }
149}

057    @Override
058    public void process(URI uri) throws Exception {
059        Map<String,String> params = URIUtil.parseQueryString(uri);
060        String server = uri.toString();
061        String metadataPrefix = params.get("metadataPrefix");
062        String set = params.get("set");
063        Date from = DateUtil.parseDateISO(params.get("from"));
064        Date until = DateUtil.parseDateISO(params.get("until"));
065        final OAIClient oaiClient = OAIClientFactory.newClient(server);
066        oaiClient.setTimeout(settings.getAsInt("timeout", 60000));
067        ListRecordsRequest request = oaiClient.newListRecordsRequest()
068                .setMetadataPrefix(metadataPrefix)
069                .setSet(set)
070                .setFrom(from, OAIDateResolution.DAY)
071                .setUntil(until, OAIDateResolution.DAY);
072        do {
073            try {
074                request.addHandler("xml".equals(settings.get("handler")) ?
075                        xmlMetadataHandler() : rdfMetadataHandler());
076                ListRecordsListener listener = new ListRecordsListener(request);
077                listener.setScrubCharacters(settings.getAsBoolean("scrubxml", true));
078                request.prepare().execute(listener).waitFor();
079                if (listener.getResponse() != null) {
080                    StringWriter w = new StringWriter();
081                    try {
082                        listener.getResponse().to(w);
083                    } catch (IOException e) {
084                        logger.error(e.getMessage(), e);
085                    } finally {
086                        if (settings.getAsBoolean("trace", false)) {
087                            logger.info(w.toString());
088                        }
089                    }
090                    request = oaiClient.resume(request, listener.getResumptionToken());
091                    TimeValue delay = settings.getAsTime("delay", null);
092                    if (delay != null) {
093                        logger.info("delay for {} seconds", delay.getSeconds());
094                        Thread.sleep(delay.millis());
095                    }
096                }
097            } catch (IOException e) {
098                logger.error(e.getMessage(), e);
099                request = null;
100            }
101        } while (request != null);
102        oaiClient.close();
103    }

105    protected MetadataHandler xmlMetadataHandler() {
106        SimpleResourceContext resourceContext = new SimpleResourceContext();
107        resourceContext.setNamespaceContext(IRINamespaceContext.getInstance());
108        RdfResourceHandler rdfResourceHandler = new RdfResourceHandler(resourceContext);
109        return new XmlMetadataHandler()
110                .setHandler(rdfResourceHandler)
111                .setResourceContext(rdfResourceHandler.resourceContext())
112                .setOutput(new ElasticOut());
113    }
114
115    protected MetadataHandler rdfMetadataHandler() {
116        SimpleResourceContext resourceContext = new SimpleResourceContext();
117        resourceContext.setNamespaceContext(IRINamespaceContext.getInstance());
118        ResourceBuilder builder = new ResourceBuilder(resourceContext);
119        RdfXmlReader reader = new RdfXmlReader().setTripleListener(builder);
120        return new XmlMetadataHandler()
121                .setHandler(reader.getHandler())
122                .setResourceContext(resourceContext)
123                .setOutput(new ElasticOut());
124    }
125
126    class XmlMetadataHandler extends MetadataHandler {
127
128        private XmlHandler handler;
129
130        private ResourceContext resourceContext;
131
132        private RdfOutput output;
133
134        public XmlMetadataHandler setHandler(XmlHandler handler) {
135            this.handler = handler;
136            this.handler.setDefaultNamespace("oai_dc","http://www.openarchives.org/OAI/2.0/oai_dc/");
137            return this;
138        }
139
140        public XmlMetadataHandler setResourceContext(ResourceContext resourceContext) {
141            this.resourceContext = resourceContext;
142            return this;
143        }
144
145        public XmlMetadataHandler setOutput(RdfOutput output) {
146            this.output = output;
147            return this;
148        }
149
150        @Override
151        public void startDocument() throws SAXException {
152            handler.startDocument();
153        }
154
155        @Override
156        public void endDocument() throws SAXException {
157            handler.endDocument();
158            String identifier = getHeader().getIdentifier();
159            try {
160                IRI iri = IRI.builder().scheme("http")
161                        .host(settings.get("index"))
162                        .query(settings.get("type"))
163                        .fragment(identifier).build();
164                resourceContext.getResource().id(iri);
165                output.output(resourceContext);
166            } catch (IOException e) {
167                logger.error(e.getMessage(), e);
168            }
169        }
170
171        @Override
172        public void startPrefixMapping(String prefix, String nsURI) throws SAXException {
173            handler.startPrefixMapping(prefix, nsURI);
174            resourceContext.getNamespaceContext().addNamespace(prefix, nsURI);
175            if ("".equals(prefix)) {
176                handler.setDefaultNamespace(settings.get("type"), nsURI);
177                resourceContext.getNamespaceContext().addNamespace(settings.get("type"), nsURI);
178            }
179        }
180
181        @Override
182        public void endPrefixMapping(String string) throws SAXException {
183            handler.endPrefixMapping(string);
184        }
185
186        @Override
187        public void startElement(String ns, String localname, String string2, Attributes atrbts) throws SAXException {
188            handler.startElement(ns, localname, string2, atrbts);
189        }
190
191        @Override
192        public void endElement(String ns, String localname, String string2) throws SAXException {
193            handler.endElement(ns, localname, string2);
194        }
195
196        @Override
197        public void characters(char[] chars, int i, int i1) throws SAXException {
198            handler.characters(chars, i, i1);
199        }
200    }

202    class ResourceBuilder implements TripleListener {
203
204        ResourceContext resourceContext;
205
206        ResourceBuilder(ResourceContext resourceContext) {
207            this.resourceContext = resourceContext;
208        }
209
210        @Override
211        public TripleListener begin() {
212            return this;
213        }
214
215        @Override
216        public TripleListener startPrefixMapping(String prefix, String uri) {
217            return this;
218        }
219
220        @Override
221        public TripleListener endPrefixMapping(String prefix) {
222            return this;
223        }
224
225        @Override
226        public ResourceBuilder newIdentifier(IRI identifier) {
227            resourceContext.setResource(resourceContext.newResource());
228            resourceContext.getResource().id(identifier);
229            return this;
230        }
231
232        @Override
233        public ResourceBuilder triple(Triple triple) {
234            resourceContext.getResource().add(triple);
235            return this;
236        }
237
238        @Override
239        public TripleListener end() {
240            return this;
241        }
242    }

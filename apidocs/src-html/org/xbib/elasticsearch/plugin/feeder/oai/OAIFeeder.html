085    public Feeder<T, R, P> beforeRun() throws IOException {
086        if (spec == null) {
087            throw new IllegalArgumentException("no spec?");
088        }
089        if (settings == null) {
090            throw new IllegalArgumentException("no settings?");
091        }
092        if (ingest == null) {
093            Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
094            Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
095                    Runtime.getRuntime().availableProcessors());
096            ByteSizeValue maxvolume = settings.getAsBytesSize("maxbulkvolume", ByteSizeValue.parseBytesSizeValue("10m"));
097            TimeValue maxrequestwait = settings.getAsTime("maxrequestwait", TimeValue.timeValueSeconds(60));
098            ingest = "ingest".equals(settings.get("client")) ?
099                    new IngestTransportClient()
100                    : new BulkTransportClient();
101            ingest.maxActionsPerBulkRequest(maxbulkactions)
102                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests)
103                    .maxVolumePerBulkRequest(maxvolume)
104                    .maxRequestWait(maxrequestwait);
105            ingest.newClient(URI.create(settings.get("elasticsearch")));
106            this.sink = new ResourceSink(ingest);
107        }
108        // create queue
109        super.beforeRun();
110        return this;
111    }

113    @Override
114    public void executeTask(Map<String, Object> map) throws Exception {
115        logger.info("{}", map);
116
117        String index = settings.get("index", "oai");
118        String type = settings.get("type", "oai");
119        if (spec.containsKey("index_settings")) {
120            try {
121                ingest.setSettings(settingsBuilder().put(new JsonSettingsLoader()
122                        .load(jsonBuilder().map((Map<String, Object>) spec.get("index_settings")).string()))
123                        .build());
124                if (spec.containsKey("type_mapping")) {
125                    ingest.addMapping(type,
126                            jsonBuilder().map((Map<String, Object>) spec.get("type_mapping")).string());
127                }
128            } catch (Exception e) {
129                logger.error(e.getMessage(),e);
130            }
131        }
132        ingest.newIndex(index);
133
134        URI uri = URI.create(map.get("url").toString());
135        Map<String, String> params = URIUtil.parseQueryString(uri);
136        String server = uri.toString();
137        String metadataPrefix = params.get("metadataPrefix");
138        String set = params.get("set");
139        Date from = "now".equals(params.get("from")) ? new Date() :
140                DateUtil.parseDateISO(params.get("from"));
141        Date until = "now".equals(params.get("until")) ? new Date() :
142                DateUtil.parseDateISO(params.get("until"));
143        final OAIClient oaiClient = OAIClientFactory.newClient(server);
144        oaiClient.setTimeout(settings.getAsInt("timeout", 60000));
145
146        ListRecordsRequest request = oaiClient.newListRecordsRequest()
147                .setMetadataPrefix(metadataPrefix)
148                .setSet(set)
149                .setFrom(from, OAIDateResolution.DAY)
150                .setUntil(until, OAIDateResolution.DAY);
151
152        do {
153            try {
154                request.addHandler("xml".equals(settings.get("handler")) ?
155                        xmlMetadataHandler() : rdfMetadataHandler());
156                ListRecordsListener listener = new ListRecordsListener(request);
157                listener.setScrubCharacters(settings.getAsBoolean("scrubxml", true));
158                request.prepare().execute(listener).waitFor();
159                if (listener.getResponse() != null) {
160                    StringWriter w = new StringWriter();
161                    try {
162                        listener.getResponse().to(w);
163                    } catch (IOException e) {
164                        logger.error(e.getMessage(), e);
165                    } finally {
166                        if (settings.getAsBoolean("trace", false)) {
167                            logger.info(w.toString());
168                        }
169                    }
170                    request = oaiClient.resume(request, listener.getResumptionToken());
171                    if (request != null && request.getResumptionToken() != null) {
172                        logger.info("{} received resumption token {}", server, request.getResumptionToken().toString());
173                    }
174                }
175            } catch (IOException e) {
176                logger.error(e.getMessage(), e);
177                request = null;
178            }
179        } while (request != null && !isInterrupted());
180        oaiClient.close();
181    }

183    protected MetadataHandler xmlMetadataHandler() {
184        IRINamespaceContext namespaceContext = IRINamespaceContext.getInstance();
185        ResourceContext<Resource> resourceContext = new SimpleResourceContext()
186                .setContentBuilder(contentBuilder(namespaceContext))
187                .setNamespaceContext(namespaceContext);
188        resourceContext.setNamespaceContext(IRINamespaceContext.getInstance());
189        RdfResourceHandler rdfResourceHandler = new RdfResourceHandler(resourceContext);
190        return new XmlMetadataHandler()
191                .setAttributeParsing(true)
192                .setHandler(rdfResourceHandler)
193                .setResourceContext(rdfResourceHandler.resourceContext())
194                .setOutput(new ElasticOut());
195    }
196
197    protected MetadataHandler rdfMetadataHandler() {
198        IRINamespaceContext namespaceContext = IRINamespaceContext.getInstance();
199        ResourceContext<Resource> resourceContext = new SimpleResourceContext()
200                .setContentBuilder(contentBuilder(namespaceContext))
201                .setNamespaceContext(IRINamespaceContext.getInstance());
202        RdfXmlReader reader = new RdfXmlReader().setTripleListener(new GeoJSONFilter(resourceContext));
203        return new XmlMetadataHandler()
204                .setAttributeParsing(false)
205                .setHandler(reader.getHandler())
206                .setResourceContext(resourceContext)
207                .setOutput(new ElasticOut());
208    }

210    protected ContentBuilder contentBuilder(IRINamespaceContext namespaceContext) {
211        return new DereferencingContentBuilder(namespaceContext,
212                ingest.client(),
213                settings.get("deref_index"),
214                settings.get("deref_type"),
215                settings.get("deref_prefix"),
216                settings.getAsArray("deref_field"));
217    }

219    class XmlMetadataHandler extends MetadataHandler {
220
221        private XmlHandler handler;
222
223        private ResourceContext resourceContext;
224
225        private RdfOutput output;
226
227        private boolean attributeParsingEnabled;
228
229        private boolean attributes;
230
231        private String uri;
232
233        private String qname;
234
235        public XmlMetadataHandler setHandler(XmlHandler handler) {
236            this.handler = handler;
237            this.handler.setDefaultNamespace("oai_dc", "http://www.openarchives.org/OAI/2.0/oai_dc/");
238            return this;
239        }
240
241        public XmlMetadataHandler setResourceContext(ResourceContext resourceContext) {
242            this.resourceContext = resourceContext;
243            return this;
244        }
245
246        public XmlMetadataHandler setAttributeParsing(boolean attributeParsingEnabled) {
247            this.attributeParsingEnabled = attributeParsingEnabled;
248            return this;
249        }
250
251        public XmlMetadataHandler setOutput(RdfOutput output) {
252            this.output = output;
253            return this;
254        }
255
256        @Override
257        public void startDocument() throws SAXException {
258            handler.startDocument();
259        }
260
261        @Override
262        public void endDocument() throws SAXException {
263            handler.endDocument();
264            String identifier = getHeader().getIdentifier();
265            try {
266                if (resourceContext.getResource() != null) {
267                    IRI iri = IRI.builder().scheme("http")
268                            .host(getSettings().get("index", "oai"))
269                            .query(getSettings().get("type", "oai"))
270                            .fragment(identifier).build();
271                    resourceContext.getResource().id(iri);
272                }
273                output.output(resourceContext);
274            } catch (IOException e) {
275                logger.error(e.getMessage(), e);
276            }
277        }
278
279        @Override
280        public void startPrefixMapping(String prefix, String nsURI) throws SAXException {
281            handler.startPrefixMapping(prefix, nsURI);
282            resourceContext.getNamespaceContext().addNamespace(prefix, nsURI);
283            if ("".equals(prefix)) {
284                handler.setDefaultNamespace(getSettings().get("type", "oai"), nsURI);
285                resourceContext.getNamespaceContext().addNamespace(getSettings().get("type", "oai"), nsURI);
286            }
287        }
288
289        @Override
290        public void endPrefixMapping(String string) throws SAXException {
291            handler.endPrefixMapping(string);
292        }
293
294        @Override
295        public void startElement(String ns, String localname, String string2, Attributes atrbts) throws SAXException {
296            handler.startElement(ns, localname, string2, atrbts);
297            attributes = false;
298            if (attributeParsingEnabled) {
299                uri = ns;
300                this.qname = string2;
301                for (int i = 0; i < atrbts.getLength(); i++) {
302                    String uri = atrbts.getURI(i);
303                    String qname = atrbts.getQName(i);
304                    char[] ch = atrbts.getValue(i).toCharArray();
305                    if (!qname.startsWith(XMLNS.NS_PREFIX) && !XSI.NS_URI.equals(uri)) {
306                        String localName = "attr_" + atrbts.getLocalName(i);
307                        int pos = qname.indexOf(':');
308                        String attr_qname = pos > 0 ? qname.substring(0, pos + 1) + localName : localName;
309                        handler.startElement(uri, localName, attr_qname, emptyAttributes);
310                        handler.characters(ch, 0, ch.length);
311                        handler.endElement(uri, localName, attr_qname);
312                        attributes = true;
313                    } else if (qname.startsWith(XMLNS.NS_PREFIX) && !XSI.NS_URI.equals(uri)) {
314                        int pos = qname.indexOf(':');
315                        String prefix = pos > 0 ? qname.substring(pos + 1) : "xmlns";
316                        handler.startElement(XMLNS.NS_URI, "_namespace", XMLNS.NS_PREFIX + ":_namespace", emptyAttributes);
317                        handler.startElement(XMLNS.NS_URI, "_prefix", XMLNS.NS_PREFIX + ":_prefix", emptyAttributes);
318                        char[] p = prefix.toCharArray();
319                        handler.characters(p, 0, p.length);
320                        handler.endElement(XMLNS.NS_URI, "_prefix", XMLNS.NS_PREFIX + ":_prefix");
321                        handler.startElement(XMLNS.NS_URI, "_uri", XMLNS.NS_PREFIX + ":_uri", emptyAttributes);
322                        handler.characters(ch, 0, ch.length);
323                        handler.endElement(XMLNS.NS_URI, "_uri", XMLNS.NS_PREFIX + ":_uri");
324                        handler.endElement(XMLNS.NS_URI, "_namespace", XMLNS.NS_PREFIX + ":_namespace");
325                        attributes = true;
326                    }
327                }
328            }
329        }
330
331        @Override
332        public void endElement(String ns, String localname, String string2) throws SAXException {
333            handler.endElement(ns, localname, string2);
334        }
335
336        @Override
337        public void characters(char[] chars, int i, int i1) throws SAXException {
338            if (attributes) {
339                handler.startElement(uri, "_value", qname, emptyAttributes);
340            }
341            handler.characters(chars, i, i1);
342            if (attributes) {
343                handler.endElement(uri, "_value", qname);
344            }
345        }
346
347        private final Attributes emptyAttributes = new AttributesImpl();
348    }

350    class ElasticOut extends RdfOutput {
351        @Override
352        public RdfOutput output(ResourceContext<Resource> resourceContext) throws IOException {
353            if (resourceContext == null) {
354                return this;
355            }
356            if (resourceContext.getResources() == null) {
357                // single document
358                sink.output(resourceContext, resourceContext.getResource(), resourceContext.getContentBuilder());
359            } else {
360                for (Resource resource : resourceContext.getResources()) {
361                    // multiple documents. Rewrite IRI for ES index/type addressing
362                    String index = getSettings().get("index", "oai");
363                    String type = getSettings().get("type", "oai");
364                    if (index.equals(resource.id().getHost())) {
365                        IRI iri = IRI.builder().scheme("http").host(index).query(type)
366                                .fragment(resource.id().getFragment()).build();
367                        resource.add("iri", resource.id().getFragment());
368                        resource.id(iri);
369                    } else {
370                        IRI iri = IRI.builder().scheme("http").host(index).query(type)
371                                .fragment(resource.id().toString()).build();
372                        resource.add("iri", resource.id().toString());
373                        resource.id(iri);
374                    }
375                    sink.output(resourceContext, resource, resourceContext.getContentBuilder());
376                }
377            }
378            return this;
379        }
380    }

382    class GeoJSONFilter implements TripleListener {
383
384        ResourceContext<Resource> resourceContext;
385
386        String lat;
387
388        String lon;
389
390        GeoJSONFilter(ResourceContext<Resource> resourceContext) {
391            this.resourceContext = resourceContext;
392        }
393
394        @Override
395        public TripleListener begin() {
396            return this;
397        }
398
399        @Override
400        public TripleListener startPrefixMapping(String prefix, String uri) {
401            return this;
402        }
403
404        @Override
405        public TripleListener endPrefixMapping(String prefix) {
406            return this;
407        }
408
409        @Override
410        public TripleListener newIdentifier(IRI identifier) {
411            return this;
412        }
413
414        @Override
415        public TripleListener triple(Triple triple) {
416            resourceContext.triple(triple);
417            if (triple.predicate().id().toString().equals(GEO_LAT.toString())) {
418                lat = triple.object().toString();
419            } else if (triple.predicate().id().toString().equals(GEO_LON.toString())) {
420                lon = triple.object().toString();
421            }
422            if (lat != null && lon != null && !"0.0".equals(lat) && !"0.0".equals(lon)) {
423                // create GeoJSON array
424                resourceContext.triple(new SimpleTriple(triple.subject(),
425                        "location",
426                        new SimpleLiteral().object(lon).type(XSD.DOUBLE)));
427                resourceContext.triple(new SimpleTriple(triple.subject(),
428                        "location",
429                        new SimpleLiteral().object(lat).type(XSD.DOUBLE)));
430                lon = null;
431                lat = null;
432            }
433            return this;
434        }
435
436        @Override
437        public TripleListener end() {
438            return this;
439        }
440    }

442    private final static IRI GEO_LAT = IRI.create("http://www.w3.org/2003/01/geo/wgs84_pos#lat");
443
444    private final static IRI GEO_LON = IRI.create("http://www.w3.org/2003/01/geo/wgs84_pos#long");

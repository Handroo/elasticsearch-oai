064    @Override
065    public void process(URI uri) throws Exception {
066        active(uri);
067        Map<String,String> params = URIUtil.parseQueryString(uri);
068        String server = uri.toString();
069        String metadataPrefix = params.get("metadataPrefix");
070        String set = params.get("set");
071        Date from = "now".equals(params.get("from")) ? new Date() :
072                DateUtil.parseDateISO(params.get("from"));
073        Date until = "now".equals(params.get("until")) ? new Date() :
074                DateUtil.parseDateISO(params.get("until"));
075        final OAIClient oaiClient = OAIClientFactory.newClient(server);
076        oaiClient.setTimeout(settings.getAsInt("timeout", 60000));
077        ListRecordsRequest request = oaiClient.newListRecordsRequest()
078                .setMetadataPrefix(metadataPrefix)
079                .setSet(set)
080                .setFrom(from, OAIDateResolution.DAY)
081                .setUntil(until, OAIDateResolution.DAY);
082        do {
083            try {
084                request.addHandler("xml".equals(settings.get("handler")) ?
085                        xmlMetadataHandler() : rdfMetadataHandler());
086                ListRecordsListener listener = new ListRecordsListener(request);
087                listener.setScrubCharacters(settings.getAsBoolean("scrubxml", true));
088                request.prepare().execute(listener).waitFor();
089                if (listener.getResponse() != null) {
090                    StringWriter w = new StringWriter();
091                    try {
092                        listener.getResponse().to(w);
093                    } catch (IOException e) {
094                        logger.error(e.getMessage(), e);
095                    } finally {
096                        if (settings.getAsBoolean("trace", false)) {
097                            logger.info(w.toString());
098                        }
099                    }
100                    request = oaiClient.resume(request, listener.getResumptionToken());
101                    if (request != null && request.getResumptionToken() != null) {
102                        logger.info("{} received resumption token {}", uri, request.getResumptionToken().toString());
103                    }
104                    TimeValue delay = settings.getAsTime("delay", null);
105                    if (delay != null) {
106                        logger.info("{} delay for {} seconds", uri, delay.getSeconds());
107                        Thread.sleep(delay.millis());
108                    }
109                }
110            } catch (IOException e) {
111                logger.error(e.getMessage(), e);
112                request = null;
113            }
114        } while (request != null && !getInterrupted());
115        inactive(uri);
116        oaiClient.close();
117    }

119    protected MetadataHandler xmlMetadataHandler() {
120        IRINamespaceContext namespaceContext = IRINamespaceContext.getInstance();
121        ResourceContext<Resource> resourceContext = new SimpleResourceContext()
122                .setContentBuilder(contentBuilder(namespaceContext))
123                .setNamespaceContext(namespaceContext);
124        resourceContext.setNamespaceContext(IRINamespaceContext.getInstance());
125        RdfResourceHandler rdfResourceHandler = new RdfResourceHandler(resourceContext);
126        return new XmlMetadataHandler()
127                .setHandler(rdfResourceHandler)
128                .setResourceContext(rdfResourceHandler.resourceContext())
129                .setOutput(new ElasticOut());
130    }
131
132    protected MetadataHandler rdfMetadataHandler() {
133        IRINamespaceContext namespaceContext = IRINamespaceContext.getInstance();
134        ResourceContext<Resource> resourceContext = new SimpleResourceContext()
135                .setContentBuilder(contentBuilder(namespaceContext))
136                .setNamespaceContext(IRINamespaceContext.getInstance());
137        RdfXmlReader reader = new RdfXmlReader().setTripleListener(new GeoJSONFilter(resourceContext));
138        return new XmlMetadataHandler()
139                .setHandler(reader.getHandler())
140                .setResourceContext(resourceContext)
141                .setOutput(new ElasticOut());
142    }

144    protected ContentBuilder contentBuilder(IRINamespaceContext namespaceContext) {
145        return new DereferencingContentBuilder<>(namespaceContext,
146                ingest.client(),
147                settings.get("deref_index"),
148                settings.get("deref_type"),
149                settings.get("deref_prefix"),
150                settings.getAsArray("deref_field"));
151    }

153    class XmlMetadataHandler extends MetadataHandler {
154
155        private XmlHandler handler;
156
157        private ResourceContext resourceContext;
158
159        private RdfOutput output;
160
161        public XmlMetadataHandler setHandler(XmlHandler handler) {
162            this.handler = handler;
163            this.handler.setDefaultNamespace("oai_dc","http://www.openarchives.org/OAI/2.0/oai_dc/");
164            return this;
165        }
166
167        public XmlMetadataHandler setResourceContext(ResourceContext resourceContext) {
168            this.resourceContext = resourceContext;
169            return this;
170        }
171
172        public XmlMetadataHandler setOutput(RdfOutput output) {
173            this.output = output;
174            return this;
175        }
176
177        @Override
178        public void startDocument() throws SAXException {
179            handler.startDocument();
180        }
181
182        @Override
183        public void endDocument() throws SAXException {
184            handler.endDocument();
185            String identifier = getHeader().getIdentifier();
186            try {
187                if (resourceContext.getResource() != null) {
188                    IRI iri = IRI.builder().scheme("http")
189                            .host(settings.get("index"))
190                            .query(settings.get("type"))
191                            .fragment(identifier).build();
192                    resourceContext.getResource().id(iri);
193                }
194                output.output(resourceContext);
195            } catch (IOException e) {
196                logger.error(e.getMessage(), e);
197            }
198        }
199
200        @Override
201        public void startPrefixMapping(String prefix, String nsURI) throws SAXException {
202            handler.startPrefixMapping(prefix, nsURI);
203            resourceContext.getNamespaceContext().addNamespace(prefix, nsURI);
204            if ("".equals(prefix)) {
205                handler.setDefaultNamespace(settings.get("type"), nsURI);
206                resourceContext.getNamespaceContext().addNamespace(settings.get("type"), nsURI);
207            }
208        }
209
210        @Override
211        public void endPrefixMapping(String string) throws SAXException {
212            handler.endPrefixMapping(string);
213        }
214
215        @Override
216        public void startElement(String ns, String localname, String string2, Attributes atrbts) throws SAXException {
217            handler.startElement(ns, localname, string2, atrbts);
218        }
219
220        @Override
221        public void endElement(String ns, String localname, String string2) throws SAXException {
222            handler.endElement(ns, localname, string2);
223        }
224
225        @Override
226        public void characters(char[] chars, int i, int i1) throws SAXException {
227            handler.characters(chars, i, i1);
228        }
229    }

231    class ElasticOut extends RdfOutput {
232        @Override
233        public RdfOutput output(ResourceContext<Resource> resourceContext) throws IOException {
234            if (resourceContext == null) {
235                return this;
236            }
237            if (resourceContext.getResources() == null) {
238                // single document
239                sink.output(resourceContext, resourceContext.getResource(), resourceContext.getContentBuilder());
240            } else for (Resource resource : resourceContext.getResources()) {
241                // multiple documents. Rewrite IRI for ES index/type addressing
242                String index = settings.get("index");
243                String type = settings.get("type");
244                if (index.equals(resource.id().getHost())) {
245                    IRI iri = IRI.builder().scheme("http").host(index).query(type)
246                            .fragment(resource.id().getFragment()).build();
247                    resource.add("iri", resource.id().getFragment());
248                    resource.id(iri);
249                } else {
250                    IRI iri = IRI.builder().scheme("http").host(index).query(type)
251                            .fragment(resource.id().toString()).build();
252                    resource.add("iri", resource.id().toString());
253                    resource.id(iri);
254                }
255                sink.output(resourceContext, resource, resourceContext.getContentBuilder());
256            }
257            return this;
258        }
259    }

261    class GeoJSONFilter implements TripleListener {
262
263        ResourceContext<Resource> resourceContext;
264
265        String lat;
266
267        String lon;
268
269        GeoJSONFilter(ResourceContext<Resource> resourceContext) {
270            this.resourceContext = resourceContext;
271        }
272
273        @Override
274        public TripleListener begin() {
275            return this;
276        }
277
278        @Override
279        public TripleListener startPrefixMapping(String prefix, String uri) {
280            return this;
281        }
282
283        @Override
284        public TripleListener endPrefixMapping(String prefix) {
285            return this;
286        }
287
288        @Override
289        public TripleListener newIdentifier(IRI identifier) {
290            return this;
291        }
292
293        @Override
294        public TripleListener triple(Triple triple) {
295            resourceContext.triple(triple);
296            if (triple.predicate().id().toString().equals(GEO_LAT.toString())) {
297                lat = triple.object().toString();
298            } else if (triple.predicate().id().toString().equals(GEO_LON.toString())) {
299                lon = triple.object().toString();
300            }
301            if (lat != null  && lon != null && !"0.0".equals(lat) && !"0.0".equals(lon)) {
302                // create GeoJSON array
303                resourceContext.triple(new SimpleTriple(triple.subject(),
304                        "location",
305                        new SimpleLiteral<>().object(lon).type(XSD.DOUBLE)));
306                resourceContext.triple(new SimpleTriple(triple.subject(),
307                        "location",
308                        new SimpleLiteral<>().object(lat).type(XSD.DOUBLE)));
309                lon = null;
310                lat = null;
311            }
312            return this;
313        }
314
315        @Override
316        public TripleListener end() {
317            return this;
318        }
319    }
320
321    private final static IRI GEO_LAT = IRI.create("http://www.w3.org/2003/01/geo/wgs84_pos#lat");
322
323    private final static IRI GEO_LON = IRI.create("http://www.w3.org/2003/01/geo/wgs84_pos#long");

042/**
043 * Base class for all feeders and feed pipelines.
044 *
045 * A feed pipeline is a sequence of URI in form of pipeline elements that can be consumed
046 * in serial or parallel manner.
047 *
048 * Variables - mostly read-only - that are common to all pipeline executions are declared
049 * as static variables.
050 *
051 * @param <T>
052 * @param <R>
053 * @param <P>
054 */
055public abstract class Feeder<T, R extends PipelineRequest, P extends Pipeline<T,R>>
056        extends AbstractPipeline<URIPipelineElement, PipelineException> implements Tool {
057
058    private final static Logger logger = LoggerFactory.getLogger(Feeder.class.getSimpleName());
059
060    /**
061     * The settings for the feeder
062     */
063    protected Settings settings;
064
065    /**
066     * The URI sources for the feed processing
067     */
068    protected Queue<URI> input;
069
070    /**
071     * The ingester routine, shared by all pipelines.
072     */
073    protected Ingest ingest;
074
075    /**
076     * Where generated RDF should be sent to.
077     */
078    protected ResourceSink sink;
079
080    /**
081     * This executor can start the feeder at scheduled times.
082     */
083    private ScheduledThreadPoolExecutor scheduledThreadPoolExecutor;
084
085    /**
086     * The executor for pipelines.
087     */
088    private MetricSimplePipelineExecutor<T,R,P> executor;
089
090    /**
091     * The state of the feeder, relevant for persistency if the feeder is running in a river
092     */
093    private RiverState riverState;
094
095    /**
096     *  A counter for the state bookkeeping.
097     */
098    private final AtomicLong counter = new AtomicLong();
099
100    /**
101     * A flag that indicates the feeder should terminate.
102     */
103    private volatile boolean interrupted;
104
105    private List<Feeder> registry = newLinkedList();
106
107    public Feeder() {
108        registry.add(this);
109    }
110
111    public Feeder(Feeder feeder) {
112        this.settings = feeder.settings;
113        this.input = feeder.input;
114        this.ingest = feeder.ingest;
115        this.sink = feeder.sink;
116        this.scheduledThreadPoolExecutor = feeder.scheduledThreadPoolExecutor;
117        this.riverState = feeder.riverState;
118        feeder.registry.add(this);
119    }
120
121    @Override
122    public void schedule(Thread thread) {
123        String[] schedule = settings.getAsArray("schedule");
124        Long seconds = settings.getAsTime("interval", TimeValue.timeValueSeconds(0)).seconds();
125        if (schedule != null && schedule.length > 0) {
126            CronThreadPoolExecutor e = new CronThreadPoolExecutor(settings.getAsInt("cronpoolsize", 4));
127            for (String cron : schedule) {
128                e.schedule(thread, new CronExpression(cron));
129            }
130            this.scheduledThreadPoolExecutor = e;
131            logger.info("scheduled with cron expressions {}", Arrays.asList(schedule));
132        } else if (seconds > 0L) {
133            this.scheduledThreadPoolExecutor = new ScheduledThreadPoolExecutor(settings.getAsInt("cronpoolsize", 4));
134            scheduledThreadPoolExecutor.scheduleAtFixedRate(thread, 0L, seconds, TimeUnit.SECONDS);
135            logger.info("scheduled at fixed rate of {} seconds", seconds);
136        } else {
137            thread.start();
138            logger.info("started");
139        }
140    }
141
142    public Feeder<T,R,P> setSettings(Settings newSettings) {
143        this.settings = newSettings;
144        return this;
145    }
146
147    public void setClient(Client c) {
148        Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
149        Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
150                Runtime.getRuntime().availableProcessors());
151        this.ingest = new NodeClient()
152                .maxActionsPerBulkRequest(maxbulkactions)
153                .maxConcurrentBulkRequests(maxconcurrentbulkrequests)
154                .newClient(c);
155    }
156
157    public Feeder<T,R,P> setRiverState(RiverState riverState) {
158        this.riverState = riverState;
159        return this;
160    }
161
162    public RiverState getRiverState() {
163        return riverState;
164    }
165
166    @Override
167    public Feeder<T,R,P> readFrom(Reader reader) {
168        this.settings = settingsBuilder().loadFromReader(reader).build();
169        return this;
170    }
171
172    @Override
173    public void run() {
174        try {
175            logger.info("starting run with settings {}", settings.getAsMap());
176            prepare();
177            this.executor = new MetricSimplePipelineExecutor<T, R, P>()
178                    .setConcurrency(settings.getAsInt("concurrency", 1))
179                    .setPipelineProvider(pipelineProvider())
180                    .prepare()
181                    .execute()
182                    .waitFor();
183        } catch (Throwable t) {
184            logger.error(t.getMessage(), t);
185        } finally {
186
187            try {
188                if (executor != null) {
189                    executor.shutdown();
190                    executor = null;
191                }
192                cleanup();
193            } catch (InterruptedException e) {
194                logger.warn("interrupted", e);
195            } catch (IOException e) {
196                logger.error(e.getMessage(), e);
197            }
198            logger.info("run completed with settings {}", settings.getAsMap());
199        }
200    }
201
202    protected Feeder<T,R,P> prepare() throws IOException {
203        Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
204        Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
205                Runtime.getRuntime().availableProcessors());
206        if (ingest == null) {
207            ingest = "ingest".equals(settings.get("client")) ?
208                      new IngestTransportClient()
209                    : new BulkTransportClient();
210            ingest.maxActionsPerBulkRequest(maxbulkactions)
211                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests);
212            ingest.newClient(URI.create(settings.get("elasticsearch")));
213        }
214        String index = settings.get("index");
215        String type = settings.get("type");
216        try {
217            ingest.setting(Feeder.class.getResourceAsStream("/" + index + "/settings"));
218        } catch (Exception e) {
219            // ignore
220        }
221        try {
222            ingest.mapping(type, Feeder.class.getResourceAsStream("/" + index + "/" + type + ".mapping"));
223        } catch (Exception e) {
224            // ignore
225        }
226        ingest.setIndex(index)
227                .setType(type)
228                .newIndex()
229                .startBulk();
230        this.sink = new ResourceSink(ingest);
231        this.input = new ConcurrentLinkedQueue<URI>();
232        String[] inputs = settings.getAsArray("input");
233        for (String uri : inputs) {
234            input.offer(URI.create(uri));
235        }
236        logger.info("input = {}", input);
237        return this;
238    }
239
240    protected Feeder<T,R,P> cleanup() throws IOException {
241        ingest.flush().stopBulk();
242        ingest.updateReplicaLevel(ingest.settings().getAsInt("index.number_of_replica", 0));
243        if (riverState != null) {
244            riverState.setActive(false).save(ingest.client());
245        }
246        return this;
247    }
248
249    /**
250     * Interrupt this feeder. The feeder pipelines will terminate as soon as the next cycle is
251     * executed.
252     * @param state true if the feeder should be interrupted
253     */
254    public void setInterrupted(boolean state) {
255        interrupted = state;
256        registry.stream().filter(feeder -> this != feeder).forEach(feeder -> {
257            feeder.setInterrupted(state);
258        });
259    }
260
261    public boolean getInterrupted() {
262        return interrupted;
263    }
264
265    @Override
266    public void close() throws IOException {
267        setInterrupted(true);
268        logger.info("closing ...");
269    }
270
271    @Override
272    public boolean hasNext() {
273        return !input.isEmpty();
274    }
275
276    @Override
277    public URIPipelineElement next() {
278        return new URIPipelineElement().set(input.poll());
279    }
280
281    @Override
282    public void newRequest(Pipeline<MeterMetric, URIPipelineElement> pipeline, URIPipelineElement request) {
283        try {
284            process(request.get());
285        } catch (Exception ex) {
286            logger.error("error while getting next input: " + ex.getMessage(), ex);
287        }
288    }
289
290    @Override
291    public void error(Pipeline<MeterMetric, URIPipelineElement> pipeline, URIPipelineElement request, PipelineException error) {
292        logger.error(error.getMessage(), error);
293    }
294
295    @Override
296    public Thread shutdownHook() {
297        return new Thread() {
298            public void run() {
299                for (Feeder feeder : registry) {
300                    try {
301                        if (feeder.scheduledThreadPoolExecutor != null) {
302                            logger.info("shutting down scheduler");
303                            feeder.scheduledThreadPoolExecutor.shutdownNow();
304                            feeder.scheduledThreadPoolExecutor = null;
305                        }
306                        if (feeder.executor != null) {
307                            logger.info("shutting down executor");
308                            feeder.executor.shutdown();
309                            feeder.executor = null;
310                        }
311                    } catch (InterruptedException e) {
312                        logger.warn("interrupted", e);
313                    } catch (IOException e) {
314                        logger.error(e.getMessage(), e);
315                    }
316                }
317                if (ingest != null) {
318                    logger.info("shutting down ingester");
319                    try {
320                        ingest.stopBulk();
321                        ingest.updateReplicaLevel(settings.getAsInt("replica", 0));
322                        ingest.shutdown();
323                    } catch (IOException e) {
324                        logger.error(e.getMessage(), e);
325                    }
326                }
327                logger.info("shutdown completed");
328            }
329        };
330    }
331
332    protected void active(URI uri) throws IOException {
333        if (riverState != null && uri != null) {
334            riverState.setName(uri.toString())
335                    .setTimestamp(new Date())
336                    .setActive(true)
337                    .setCounter(counter.incrementAndGet())
338                    .save(ingest.client());
339        }
340    }
341
342    protected void inactive(URI uri) throws IOException {
343        if (riverState != null && uri != null) {
344            riverState.setName(uri.toString())
345                    .setTimestamp(new Date())
346                    .setActive(false)
347                    .save(ingest.client());
348        }
349    }
350
351    protected abstract PipelineProvider<P> pipelineProvider();
352
353    protected abstract void process(URI uri) throws Exception;

001package org.xbib.elasticsearch.plugin.river.oai;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.common.settings.loader.JsonSettingsLoader;
007import org.elasticsearch.common.util.concurrent.EsExecutors;
008import org.elasticsearch.river.AbstractRiverComponent;
009import org.elasticsearch.river.RiverIndexName;
010import org.elasticsearch.river.RiverName;
011import org.elasticsearch.river.RiverSettings;
012import org.xbib.elasticsearch.action.river.execute.RunnableRiver;
013import org.xbib.elasticsearch.action.river.state.RiverState;
014import org.xbib.elasticsearch.action.river.state.StatefulRiver;
015import org.xbib.elasticsearch.plugin.feeder.Feeder;
016import org.xbib.elasticsearch.plugin.feeder.oai.OAIFeeder;
017
018import java.io.IOException;
019import java.util.Date;
020import java.util.Map;
021
022import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
023import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
024
025/**
026 * Open Archive Initiative river
027 * The river is a thin wrapper around the OAI feeder.
028 */
029public class OAIRiver extends AbstractRiverComponent implements RunnableRiver, StatefulRiver {
030
031    private volatile Thread riverThread;
032
033    private final Feeder feeder;
034
035    private final Client client;
036
037    private volatile boolean closed;
038
039    @Inject
040    public OAIRiver(RiverName riverName, RiverSettings riverSettings,
041                    @RiverIndexName String riverIndexName, Client client) {
042        super(riverName, riverSettings);
043        if (!riverSettings.settings().containsKey("oai")) {
044            throw new IllegalArgumentException("no 'oai' settings in river settings?");
045        }
046        this.client = client;
047        this.feeder = createFeeder(riverSettings);
048    }
049
050    private Feeder createFeeder(RiverSettings riverSettings) {
051        Feeder feeder = null;
052        try {
053            Map<String, Object> spec = (Map<String, Object>) riverSettings.settings().get("oai");
054            Map<String, String> loadedSettings = new JsonSettingsLoader().load(jsonBuilder().map(spec).string());
055            Settings mySettings = settingsBuilder().put(loadedSettings).build();
056            feeder = new OAIFeeder();
057            feeder.setType("oai").setSpec(spec).setSettings(mySettings);
058        } catch (IOException e) {
059            logger.error(e.getMessage(), e);
060        }
061        return feeder;
062    }
063
064    @Override
065    public void start() {
066        feeder.setClient(client);
067        feeder.setRiverState(new RiverState()
068                        .setEnabled(true)
069                        .setStarted(new Date())
070                        .setName(riverName.getName())
071                        .setType(riverName.getType())
072                        .setCoordinates("_river", riverName.getName(), "_custom")
073        );
074        this.riverThread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
075                "river(" + riverName().getType() + "/" + riverName().getName() + ")")
076                .newThread(feeder);
077        feeder.schedule(riverThread);
078    }
079
080    @Override
081    public void close() {
082        if (closed) {
083            return;
084        }
085        closed = true;
086        logger.info("closing river({}/{})", riverName.getType(), riverName.getName());
087        try {
088            feeder.getRiverState().setEnabled(false).save(client);
089            feeder.shutdown();
090        } catch (IOException e) {
091            logger.error(e.getMessage(), e);
092        }
093        if (riverThread != null) {
094            riverThread.interrupt();
095        }
096    }
097
098    @Override
099    public RiverState getRiverState() {
100        return feeder.getRiverState();
101    }
102
103    @Override
104    public void run() {
105        feeder.run();
106    }
107
108}

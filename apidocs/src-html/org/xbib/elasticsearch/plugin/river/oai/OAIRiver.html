002package org.xbib.elasticsearch.plugin.river.oai;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.util.concurrent.EsExecutors;
007import org.elasticsearch.common.xcontent.support.XContentMapValues;
008import org.elasticsearch.river.AbstractRiverComponent;
009import org.elasticsearch.river.River;
010import org.elasticsearch.river.RiverIndexName;
011import org.elasticsearch.river.RiverName;
012import org.elasticsearch.river.RiverSettings;
013
014import org.xbib.common.settings.Settings;
015import org.xbib.elasticsearch.plugin.feeder.oai.OAIFeeder;
016
017import java.io.IOException;
018import java.util.Map;
019
020import static org.xbib.common.settings.ImmutableSettings.settingsBuilder;
021
022public class OAIRiver extends AbstractRiverComponent implements River {
023
024    private volatile Thread harvesterThread;
025
026    private final OAIFeeder feeder;
027
028    private final Client client;
029
030    @Inject
031    public OAIRiver(RiverName riverName, RiverSettings riverSettings,
032                    @RiverIndexName String riverIndexName, Client client) {
033        super(riverName, riverSettings);
034        this.client = client;
035        if (!riverSettings.settings().containsKey("oai")) {
036            throw new IllegalArgumentException("no 'oai' settings in river settings?");
037        }
038        Map<String, Object> oaiSettings = (Map<String, Object>) settings.settings().get("oai");
039
040        Settings settings = settingsBuilder()
041                .putArray("input", XContentMapValues.extractRawValues("input", oaiSettings))
042                .put("concurrency", XContentMapValues.nodeIntegerValue(oaiSettings.get("concurrency"), 1))
043                .put("handler", XContentMapValues.nodeStringValue(oaiSettings.get("handler"), "xml"))
044                .put("index", XContentMapValues.nodeStringValue(oaiSettings.get("index"), "oai"))
045                .put("type", XContentMapValues.nodeStringValue(oaiSettings.get("type"), "oai"))
046                .put("shards", XContentMapValues.nodeIntegerValue(oaiSettings.get("shards"), 1))
047                .put("replica", XContentMapValues.nodeIntegerValue(oaiSettings.get("replica"), 0))
048                .put("maxbulkactions", XContentMapValues.nodeIntegerValue(oaiSettings.get("maxbulkactions"), 1000))
049                .put("maxconcurrentbulkrequests", XContentMapValues.nodeIntegerValue(oaiSettings.get("maxconcurrentbulkrequests"), 10))
050                .put("trace", XContentMapValues.nodeBooleanValue(oaiSettings.get("trace"), false))
051                .put("scrubxml", XContentMapValues.nodeBooleanValue(oaiSettings.get("scrubxml"), true))
052                .build();
053
054        feeder = new OAIFeeder();
055        feeder.settings(settings);
056    }
057
058    @Override
059    public void start() {
060        feeder.setClient(client);
061        harvesterThread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
062                "oai_river(" + riverName().name() + ")").newThread(feeder);
063        harvesterThread.start();
064    }
065
066    @Override
067    public void close() {
068        logger.info("closing oai_river({})", riverName.name());
069        try {
070            feeder.close();
071        } catch (IOException e) {
072            logger.error(e.getMessage(), e);
073        }
074        harvesterThread.interrupt();
075    }
076
077}
